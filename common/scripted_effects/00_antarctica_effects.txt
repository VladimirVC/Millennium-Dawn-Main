# Usage
# set_temp_variable = { station_id = x }
# antarctica_setup_research_station = yes
antarctica_setup_research_station = {
	if = {
		limit = {
			meta_trigger = {
				text = {
					NOT = { has_global_flag = research_station_[ID]_exists }
				}
				ID = "[?station_id]"
			}
		}
		meta_effect = {
			text = {
				set_global_flag = research_station_[ID]_exists
				set_country_flag = research_station_[ID]_controller_@THIS
				set_variable = { global.research_station_[ID]_controller = THIS.id }
			}
			ID = "[?station_id]"
		}

		add_to_array = { country_owned_station_ids = station_id }

		if = {
			limit = { NOT = { has_country_flag = is_antarctic_research_station_constructed } }
			set_country_flag = is_antarctic_research_station_constructed
		}

		set_country_flag = antarctica_has_active_station

		set_temp_variable = { station_region = 3 }
		if = {
			limit = {
				OR = {
					capital_scope = { is_on_continent = north_america }
					capital_scope = { is_on_continent = south_america }
				}
			}
			set_temp_variable = { station_region = 1 }
		}
		else_if = {
			limit = {
				OR = {
					capital_scope = { is_on_continent = asia }
					capital_scope = { is_on_continent = australia }
				}
			}
			set_temp_variable = { station_region = 2 }
		}
		else_if = {
			limit = {
				OR = {
					capital_scope = { is_on_continent = africa }
					capital_scope = { is_on_continent = middle_east }
				}
			}
			set_temp_variable = { station_region = 4 }
		}

		meta_effect = {
			text = {
				set_variable = { global.research_station_[ID]_region_filter = [REGION] }
			}
			ID = "[?station_id]"
			REGION = "[?station_region]"
		}

		if = {
			limit = { check_variable = { selected_antarctica_station = 0 } }
			set_variable = { selected_antarctica_station = station_id }
			set_variable = { selected_antarctica_station_controller = THIS.id }
		}
		log = "[GetDateText]: [Root.GetName]: Set up research station in Antarctica. Station ID: [?station_id] with controller [Root.GetName]."
	}
	else = {
		log = "[GetDateText]: [Root.GetName]: Failed to set up research station in Antarctica. Station ID: [?station_id] already exists."
	}
}

# Usage
# set_temp_variable = { station_id = id }
# antarctica_recalculate_station_module_outputs = yes
antarctica_recalculate_station_module_outputs = {
	if = {
		limit = {
			meta_trigger = {
				text = { has_global_flag = research_station_[ID]_exists }
				ID = "[?station_id]"
			}
		}

		meta_effect = {
			text = {
				set_temp_variable = { pwr_module = global.research_station_[ID]_module_powerplant_active }
				set_temp_variable = { life_module = global.research_station_[ID]_module_life_support_active }
				set_temp_variable = { fuel_module = global.research_station_[ID]_module_fuel_storage_active }
				set_temp_variable = { house_module = global.research_station_[ID]_module_housing_active }
				set_temp_variable = { lab_module = global.research_station_[ID]_module_laboratories_active }
			}
			ID = "[?station_id]"
		}

		if = {
			limit = { check_variable = { pwr_module < 1 } }
			meta_effect = {
				text = {
					set_temp_variable = { pwr_module = global.research_station_[ID]_module_powerplant_list^0 }
				}
				ID = "[?station_id]"
			}
		}

		if = {
			limit = { check_variable = { life_module < 1 } }
			meta_effect = {
				text = {
					set_temp_variable = { life_module = global.research_station_[ID]_module_life_support_list^0 }
				}
				ID = "[?station_id]"
			}
		}

		if = {
			limit = { check_variable = { fuel_module < 1 } }
			meta_effect = {
				text = {
					set_temp_variable = { fuel_module = global.research_station_[ID]_module_fuel_storage_list^0 }
				}
				ID = "[?station_id]"
			}
		}

		if = {
			limit = { check_variable = { house_module < 1 } }
			meta_effect = {
				text = {
					set_temp_variable = { house_module = global.research_station_[ID]_module_housing_list^0 }
				}
				ID = "[?station_id]"
			}
		}

		if = {
			limit = { check_variable = { lab_module < 1 } }
			meta_effect = {
				text = {
					set_temp_variable = { lab_module = global.research_station_[ID]_module_laboratories_list^0 }
				}
				ID = "[?station_id]"
			}
		}

		set_temp_variable = { out_power = 0 }
		set_temp_variable = { out_power_total = 0 }
		set_temp_variable = { out_fuel_usage = 0 }
		set_temp_variable = { out_fuel_usage_bonus = 0 }
		set_temp_variable = { out_staff_satisfaction = 0 }
		set_temp_variable = { out_storage = 0 }
		set_temp_variable = { out_capacity = 0 }
		set_temp_variable = { out_install_cost = 0 }
		set_temp_variable = { out_iteration_speed_multiplier = 1 }
		# Reliability is module-specific and not aggregated into a station-wide value.
		set_temp_variable = { out_reliability = 0 }
		set_temp_variable = { out_construction_time = 0 }
		set_temp_variable = { out_upkeep_cost = 0 }
		set_temp_variable = { out_required_staff = 0 }
		set_temp_variable = { out_total_staff = 0 }

		# Aggregate all installed module contributions for station outputs.
		set_temp_variable = { has_powerplant_modules = 0 }
		set_temp_variable = { has_life_support_modules = 0 }
		set_temp_variable = { has_fuel_storage_modules = 0 }
		set_temp_variable = { has_housing_modules = 0 }
		set_temp_variable = { has_laboratory_modules = 0 }

		meta_effect = {
			text = {
				if = {
					limit = {
						check_variable = { global.research_station_[ID]_module_slot_2 > 0 }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_2_blizzard_damaged > 0 } }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_2_repair_days_remaining > 0 } }
					}
					set_temp_variable = { pwr_module = global.research_station_[ID]_module_slot_2 }
					set_temp_variable = { has_powerplant_modules = 1 }
					antarctica_apply_single_powerplant_station_output = yes
				}
				if = {
					limit = {
						check_variable = { global.research_station_[ID]_module_slot_11 > 0 }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_11_blizzard_damaged > 0 } }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_11_repair_days_remaining > 0 } }
					}
					set_temp_variable = { pwr_module = global.research_station_[ID]_module_slot_11 }
					set_temp_variable = { has_powerplant_modules = 1 }
					antarctica_apply_single_powerplant_station_output = yes
				}
				if = {
					limit = {
						check_variable = { global.research_station_[ID]_module_slot_13 > 0 }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_13_blizzard_damaged > 0 } }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_13_repair_days_remaining > 0 } }
					}
					set_temp_variable = { pwr_module = global.research_station_[ID]_module_slot_13 }
					set_temp_variable = { has_powerplant_modules = 1 }
					antarctica_apply_single_powerplant_station_output = yes
				}
			}
			ID = "[?station_id]"
		}

		meta_effect = {
			text = {
				for_each_loop = {
					array = global.research_station_[ID]_module_life_support_list
					value = current_module
					if = {
						limit = { check_variable = { current_module > 0 } }
						set_temp_variable = { life_module = current_module }
						set_temp_variable = { has_life_support_modules = 1 }
						antarctica_apply_single_life_support_station_output = yes
					}
				}
			}
			ID = "[?station_id]"
		}

		if = {
			limit = {
				check_variable = { has_life_support_modules < 1 }
				check_variable = { life_module > 0 }
			}
			antarctica_apply_single_life_support_station_output = yes
		}

		meta_effect = {
			text = {
				for_each_loop = {
					array = global.research_station_[ID]_module_fuel_storage_list
					value = current_module
					if = {
						limit = { check_variable = { current_module > 0 } }
						set_temp_variable = { fuel_module = current_module }
						set_temp_variable = { has_fuel_storage_modules = 1 }
						antarctica_apply_single_fuel_storage_station_output = yes
					}
				}
			}
			ID = "[?station_id]"
		}

		if = {
			limit = {
				check_variable = { has_fuel_storage_modules < 1 }
				check_variable = { fuel_module > 0 }
			}
			antarctica_apply_single_fuel_storage_station_output = yes
		}

		meta_effect = {
			text = {
				for_each_loop = {
					array = global.research_station_[ID]_module_housing_list
					value = current_module
					if = {
						limit = { check_variable = { current_module > 0 } }
						set_temp_variable = { house_module = current_module }
						set_temp_variable = { has_housing_modules = 1 }
						antarctica_apply_single_housing_station_output = yes
					}
				}
			}
			ID = "[?station_id]"
		}

		if = {
			limit = {
				check_variable = { has_housing_modules < 1 }
				check_variable = { house_module > 0 }
			}
			antarctica_apply_single_housing_station_output = yes
		}

		# Laboratory output stats aggregate all installed laboratory modules.
		meta_effect = {
			text = {
				if = {
					limit = {
						check_variable = { global.research_station_[ID]_module_slot_6 > 0 }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_6_blizzard_damaged > 0 } }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_6_repair_days_remaining > 0 } }
					}
					set_temp_variable = { has_laboratory_modules = 1 }
					set_temp_variable = { lab_module = global.research_station_[ID]_module_slot_6 }
					antarctica_apply_single_laboratory_station_output = yes
				}
				if = {
					limit = {
						check_variable = { global.research_station_[ID]_module_slot_7 > 0 }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_7_blizzard_damaged > 0 } }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_7_repair_days_remaining > 0 } }
					}
					set_temp_variable = { has_laboratory_modules = 1 }
					set_temp_variable = { lab_module = global.research_station_[ID]_module_slot_7 }
					antarctica_apply_single_laboratory_station_output = yes
				}
				if = {
					limit = {
						check_variable = { global.research_station_[ID]_module_slot_8 > 0 }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_8_blizzard_damaged > 0 } }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_8_repair_days_remaining > 0 } }
					}
					set_temp_variable = { has_laboratory_modules = 1 }
					set_temp_variable = { lab_module = global.research_station_[ID]_module_slot_8 }
					antarctica_apply_single_laboratory_station_output = yes
				}
				if = {
					limit = {
						check_variable = { global.research_station_[ID]_module_slot_9 > 0 }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_9_blizzard_damaged > 0 } }
						NOT = { check_variable = { global.research_station_[ID]_module_slot_9_repair_days_remaining > 0 } }
					}
					set_temp_variable = { has_laboratory_modules = 1 }
					set_temp_variable = { lab_module = global.research_station_[ID]_module_slot_9 }
					antarctica_apply_single_laboratory_station_output = yes
				}
			}
			ID = "[?station_id]"
		}

		if = {
			limit = {
				check_variable = { has_laboratory_modules < 1 }
				check_variable = { lab_module > 0 }
			}
			antarctica_apply_single_laboratory_station_output = yes
		}

		meta_effect = {
			text = {
				set_temp_variable = { existing_storage = global.research_station_[ID]_output_storage }
			}
			ID = "[?station_id]"
		}

		# out_storage is the module-defined max fuel capacity.
		set_temp_variable = { out_capacity = out_storage }
		set_temp_variable = { effective_storage = out_capacity }

		# Preserve current remaining fuel between recalculations when available.
		if = {
			limit = { check_variable = { existing_storage > 0 } }
			set_temp_variable = { effective_storage = existing_storage }
		}

		if = {
			limit = { check_variable = { effective_storage > out_capacity } }
			set_temp_variable = { effective_storage = out_capacity }
		}

		if = {
			limit = { check_variable = { effective_storage < 0 } }
			set_temp_variable = { effective_storage = 0 }
		}

		# Fuel efficiency bonuses reduce effective fuel usage.
		if = {
			limit = {
				check_variable = { out_fuel_usage_bonus > 0 }
				check_variable = { out_fuel_usage > 0 }
			}
			set_temp_variable = { fuel_usage_multiplier = 1 }
			subtract_from_temp_variable = { fuel_usage_multiplier = out_fuel_usage_bonus }
			clamp_temp_variable = { var = fuel_usage_multiplier min = 0 }
			multiply_temp_variable = { out_fuel_usage = fuel_usage_multiplier }
		}

		add_to_temp_variable = { out_iteration_speed_multiplier = out_staff_satisfaction }
		clamp_temp_variable = { var = out_iteration_speed_multiplier min = 0 }

		set_temp_variable = { out_fuel_capacity_days = 0 }
		if = {
			limit = { check_variable = { out_fuel_usage > 0 } }
			set_temp_variable = { out_fuel_capacity_days = effective_storage }
			divide_temp_variable = { out_fuel_capacity_days = out_fuel_usage }
			round_variable = out_fuel_capacity_days
		}

		meta_effect = {
			text = {
				set_variable = { global.research_station_[ID]_output_power = [PWR] }
				set_variable = { global.research_station_[ID]_output_power_total = [TOTAL_PWR] }
				set_variable = { global.research_station_[ID]_output_fuel_usage = [FUEL] }
				set_variable = { global.research_station_[ID]_output_fuel_capacity_days = [FUEL_DAYS] }
				set_variable = { global.research_station_[ID]_output_fuel_usage_bonus = [FUEL_BONUS] }
				set_variable = { global.research_station_[ID]_output_staff_satisfaction = [SAT] }
				set_variable = { global.research_station_[ID]_output_iteration_speed_multiplier = [ITER_MOD] }
				set_variable = { global.research_station_[ID]_output_storage = [STO] }
				set_variable = { global.research_station_[ID]_output_capacity = [CAP] }
				set_variable = { global.research_station_[ID]_output_install_cost = [INS] }
				set_variable = { global.research_station_[ID]_output_reliability = [REL] }
				set_variable = { global.research_station_[ID]_output_construction_time = [CON] }
				set_variable = { global.research_station_[ID]_output_upkeep_cost = [UPK] }
				set_variable = { global.research_station_[ID]_output_required_staff = [REQ] }
				set_variable = { global.research_station_[ID]_output_total_staff = [TOTAL_STAFF] }
			}
			ID = "[?station_id]"
			PWR = "[?out_power]"
			FUEL = "[?out_fuel_usage]"
			FUEL_DAYS = "[?out_fuel_capacity_days]"
			FUEL_BONUS = "[?out_fuel_usage_bonus]"
			TOTAL_PWR = "[?out_power_total]"
			SAT = "[?out_staff_satisfaction]"
			ITER_MOD = "[?out_iteration_speed_multiplier]"
			STO = "[?effective_storage]"
			CAP = "[?out_capacity]"
			INS = "[?out_install_cost]"
			REL = "[?out_reliability]"
			CON = "[?out_construction_time]"
			UPK = "[?out_upkeep_cost]"
			REQ = "[?out_required_staff]"
			TOTAL_STAFF = "[?out_total_staff]"
		}
	}
}

# Usage
# set_temp_variable = { station_id = id }
# set_temp_variable = { module_category = n } # 1 powerplant, 2 life support, 3 fuel storage, 4 housing, 5 laboratories
# set_temp_variable = { module_id = n }
# antarctica_set_station_module = yes
antarctica_set_station_module = {
	if = {
		limit = {
			meta_trigger = {
				text = { has_global_flag = research_station_[ID]_exists }
				ID = "[?station_id]"
			}
		}

		if = {
			limit = { check_variable = { module_category = 1 } }
			meta_effect = {
				text = {
					add_to_array = { global.research_station_[ID]_module_powerplant_list = [MOD] }
					set_variable = { global.research_station_[ID]_module_powerplant_active = [MOD] }
					set_variable = { global.research_station_[ID]_module_powerplant = [MOD] }
					set_variable = { global.research_station_[ID]_module_slot_11 = [MOD] }
				}
				ID = "[?station_id]"
				MOD = "[?module_id]"
			}
		}
		else_if = {
			limit = { check_variable = { module_category = 2 } }
			meta_effect = {
				text = {
					add_to_array = { global.research_station_[ID]_module_life_support_list = [MOD] }
					set_variable = { global.research_station_[ID]_module_life_support_active = [MOD] }
					set_variable = { global.research_station_[ID]_module_life_support = [MOD] }
					set_variable = { global.research_station_[ID]_module_slot_12 = [MOD] }
				}
				ID = "[?station_id]"
				MOD = "[?module_id]"
			}
		}
		else_if = {
			limit = { check_variable = { module_category = 3 } }
			meta_effect = {
				text = {
					add_to_array = { global.research_station_[ID]_module_fuel_storage_list = [MOD] }
					set_variable = { global.research_station_[ID]_module_fuel_storage_active = [MOD] }
					set_variable = { global.research_station_[ID]_module_fuel_storage = [MOD] }
					set_variable = { global.research_station_[ID]_module_slot_14 = [MOD] }
				}
				ID = "[?station_id]"
				MOD = "[?module_id]"
			}
		}
		else_if = {
			limit = { check_variable = { module_category = 4 } }
			meta_effect = {
				text = {
					add_to_array = { global.research_station_[ID]_module_housing_list = [MOD] }
					set_variable = { global.research_station_[ID]_module_housing_active = [MOD] }
					set_variable = { global.research_station_[ID]_module_housing = [MOD] }
					set_variable = { global.research_station_[ID]_module_slot_5 = [MOD] }
				}
				ID = "[?station_id]"
				MOD = "[?module_id]"
			}
		}
		else_if = {
			limit = { check_variable = { module_category = 5 } }
			meta_effect = {
				text = {
					add_to_array = { global.research_station_[ID]_module_laboratories_list = [MOD] }
					set_variable = { global.research_station_[ID]_module_laboratories_active = [MOD] }
					set_variable = { global.research_station_[ID]_module_slot_8 = [MOD] }
				}
				ID = "[?station_id]"
				MOD = "[?module_id]"
			}
		}

		antarctica_recalculate_station_module_outputs = yes

antarctica_process_atcp_applications = {
	for_each_scope_loop = {
		array = global.at_members

		if = {
			limit = {
				has_country_flag = antarctica_atcp_application_pending
				has_country_flag = {
					flag = antarctica_atcp_application_pending
					days > 90
				}
			}

			clr_country_flag = antarctica_atcp_application_pending
			set_country_flag = is_antarctica_atcp_consultative_member

			if = {
				limit = { NOT = { is_in_array = { global.atcp_members = THIS } } }
				add_to_array = { global.atcp_members = THIS }
			}

			log = "[GetDateText]: [Root.GetName]: Granted ATCP consultative status."
		}
	}
}

antarctica_calculate_selected_preview_module_install_time = {

	set_temp_variable = { out_required_staff = 0 }

		limit = { check_variable = { selected_antarctica_station_selected_slot_installed_module > 0 } }
		if = {
			limit = { check_variable = { selected_antarctica_station_selected_module_category = 1 } }
			set_temp_variable = { pwr_module = selected_antarctica_station_selected_slot_installed_module }
			antarctica_apply_single_powerplant_station_output = yes
		}
		else_if = {
			limit = { check_variable = { selected_antarctica_station_selected_module_category = 2 } }
			set_temp_variable = { life_module = selected_antarctica_station_selected_slot_installed_module }
			antarctica_apply_single_life_support_station_output = yes
		}
		else_if = {
			limit = { check_variable = { selected_antarctica_station_selected_module_category = 3 } }
			set_temp_variable = { fuel_module = selected_antarctica_station_selected_slot_installed_module }
			antarctica_apply_single_fuel_storage_station_output = yes
		}
		else_if = {
			limit = { check_variable = { selected_antarctica_station_selected_module_category = 4 } }
			set_temp_variable = { house_module = selected_antarctica_station_selected_slot_installed_module }
			antarctica_apply_single_housing_station_output = yes
		}
		else_if = {
			limit = { check_variable = { selected_antarctica_station_selected_module_category = 5 } }
			set_temp_variable = { lab_module = selected_antarctica_station_selected_slot_installed_module }
			antarctica_apply_single_laboratory_station_output = yes
		}

		set_temp_variable = { installed_module_power = out_power }
		set_temp_variable = { installed_module_required_staff = out_required_staff }
	}

	set_temp_variable = { out_construction_time = 0 }
	set_temp_variable = { out_power = 0 }
	set_temp_variable = { out_power_total = 0 }
	set_temp_variable = { out_required_staff = 0 }
	set_temp_variable = { out_install_cost = 0 }

	if = {
		limit = { check_variable = { selected_antarctica_station_selected_module_category = 1 } }
		set_temp_variable = { pwr_module = selected_antarctica_station_selected_module }
		antarctica_apply_single_powerplant_station_output = yes
	}
	else_if = {
		limit = { check_variable = { selected_antarctica_station_selected_module_category = 2 } }
		set_temp_variable = { life_module = selected_antarctica_station_selected_module }
		antarctica_apply_single_life_support_station_output = yes
	}
	else_if = {
		limit = { check_variable = { selected_antarctica_station_selected_module_category = 3 } }
		set_temp_variable = { fuel_module = selected_antarctica_station_selected_module }
		antarctica_apply_single_fuel_storage_station_output = yes
	}
	else_if = {
		limit = { check_variable = { selected_antarctica_station_selected_module_category = 4 } }
		set_temp_variable = { house_module = selected_antarctica_station_selected_module }
		antarctica_apply_single_housing_station_output = yes
	}
	else_if = {
		limit = { check_variable = { selected_antarctica_station_selected_module_category = 5 } }
		set_temp_variable = { lab_module = selected_antarctica_station_selected_module }
		antarctica_apply_single_laboratory_station_output = yes
	}

	# Apply seasonal multiplier (April-October = winter = 2x)
	set_temp_variable = { install_time_multiplier = 1 }
	set_temp_variable = { install_cost_multiplier = 1 }
	if = {
		limit = { OR = { check_variable = { global.month = 4 } check_variable = { global.month = 5 } check_variable = { global.month = 6 } check_variable = { global.month = 7 } check_variable = { global.month = 8 } check_variable = { global.month = 9 } check_variable = { global.month = 10 } } }
		set_temp_variable = { install_time_multiplier = 2 }
		set_temp_variable = { install_cost_multiplier = 2 }
	}

	multiply_temp_variable = { out_construction_time = install_time_multiplier }

	set_variable = { selected_antarctica_station_selected_module_install_time = out_construction_time }
	set_variable = { selected_antarctica_station_selected_module_install_time_multiplier = install_time_multiplier }
	round_variable = selected_antarctica_station_selected_module_install_time
	clamp_variable = { var = selected_antarctica_station_selected_module_install_time min = 1 max = 999 }

	# Calculate installation cost: use actual module cost (negative value), convert to positive, and apply multiplier
	# out_install_cost is negative (e.g., -0.5), so multiply by -1 to get positive cost
	set_temp_variable = { install_cost = out_install_cost }
	multiply_temp_variable = { install_cost = -1 }
	multiply_temp_variable = { install_cost = install_cost_multiplier }
	set_variable = { selected_antarctica_station_selected_module_install_cost = install_cost }

	# Calculate preview balance as a swap result: current value minus removed module plus selected module.
	set_variable = { selected_antarctica_station_preview_output_power = selected_antarctica_station_output_power }
	subtract_from_variable = { selected_antarctica_station_preview_output_power = installed_module_power }
	add_to_variable = { selected_antarctica_station_preview_output_power = out_power }

	set_variable = { selected_antarctica_station_preview_output_required_staff = selected_antarctica_station_output_required_staff }
	subtract_from_variable = { selected_antarctica_station_preview_output_required_staff = installed_module_required_staff }
	add_to_variable = { selected_antarctica_station_preview_output_required_staff = out_required_staff }
}

antarctica_process_station_fuel_consumption = {
	if = {
		limit = { NOT = { has_global_flag = antarctica_fuel_storage_migrated_v2 } }
		for_each_loop = {
			array = global.antarctica_station_ids
			value = migrate_station_id

			if = {
				limit = {
					check_variable = { migrate_station_id > 0 }
					meta_trigger = {
						text = { has_global_flag = research_station_[ID]_exists }
						ID = "[?migrate_station_id]"
					}
				}

				set_temp_variable = { station_id = migrate_station_id }
				antarctica_recalculate_station_module_outputs = yes

				meta_effect = {
					text = {
						set_temp_variable = { migrate_capacity = global.research_station_[ID]_output_capacity }
						set_temp_variable = { migrate_usage = global.research_station_[ID]_output_fuel_usage }
					}
					ID = "[?migrate_station_id]"
				}

				if = {
					limit = {
						check_variable = { migrate_capacity > 0 }
						check_variable = { migrate_usage > 0 }
					}
					set_temp_variable = { migrate_days = migrate_capacity }
					divide_temp_variable = { migrate_days = migrate_usage }
					round_variable = migrate_days

					meta_effect = {
						text = {
							set_variable = { global.research_station_[ID]_output_storage = [STORAGE] }
							set_variable = { global.research_station_[ID]_output_fuel_capacity_days = [DAYS] }
							set_global_flag = research_station_[ID]_fuel_initialized
						}
						ID = "[?migrate_station_id]"
						STORAGE = "[?migrate_capacity]"
						DAYS = "[?migrate_days]"
					}
				}
			}
		}

		set_global_flag = antarctica_fuel_storage_migrated_v2
	}

	for_each_loop = {
		array = global.antarctica_station_ids
		value = process_station_id

		if = {
			limit = {
				check_variable = { process_station_id > 0 }
				meta_trigger = {
					text = { has_global_flag = research_station_[ID]_exists }
					ID = "[?process_station_id]"
				}
			}

			meta_effect = {
				text = {
					set_temp_variable = { fuel_usage = global.research_station_[ID]_output_fuel_usage }
					set_temp_variable = { current_storage = global.research_station_[ID]_output_storage }
					set_temp_variable = { max_capacity = global.research_station_[ID]_output_capacity }
				}
				ID = "[?process_station_id]"
			}

			if = {
				limit = { check_variable = { max_capacity < 1 } }
				set_temp_variable = { station_id = process_station_id }
				antarctica_recalculate_station_module_outputs = yes
				meta_effect = {
					text = {
						set_temp_variable = { fuel_usage = global.research_station_[ID]_output_fuel_usage }
						set_temp_variable = { current_storage = global.research_station_[ID]_output_storage }
						set_temp_variable = { max_capacity = global.research_station_[ID]_output_capacity }
					}
					ID = "[?process_station_id]"
				}
			}

			# One-time initialization for legacy saves: if storage was never seeded,
			# start from max capacity once and then let daily consumption deplete it.
			if = {
				limit = {
					check_variable = { current_storage < 1 }
					check_variable = { max_capacity > 0 }
					meta_trigger = {
						text = {
							NOT = { has_global_flag = research_station_[ID]_fuel_initialized }
						}
						ID = "[?process_station_id]"
					}
				}
				set_temp_variable = { current_storage = max_capacity }
				meta_effect = {
					text = { set_global_flag = research_station_[ID]_fuel_initialized }
					ID = "[?process_station_id]"
				}
			}

			if = {
				limit = { check_variable = { fuel_usage > 0 } }
				subtract_from_temp_variable = { current_storage = fuel_usage }

				if = {
					limit = { check_variable = { current_storage < 0 } }
					set_temp_variable = { current_storage = 0 }
				}

				set_temp_variable = { fuel_capacity_days = current_storage }
				divide_temp_variable = { fuel_capacity_days = fuel_usage }
				round_variable = fuel_capacity_days

				meta_effect = {
					text = {
						set_variable = { global.research_station_[ID]_output_storage = [STORAGE] }
						set_variable = { global.research_station_[ID]_output_fuel_capacity_days = [DAYS] }
					}
					ID = "[?process_station_id]"
					STORAGE = "[?current_storage]"
					DAYS = "[?fuel_capacity_days]"
				}
			}
		}
	}

	if = {
		limit = { check_variable = { selected_antarctica_station > 0 } }
		antarctica_set_selected_station_controller = yes
	}
}
antarctica_process_station_research_progress = {
	for_each_loop = {
		array = global.antarctica_station_ids
		value = process_station_id

		if = {
			limit = {
				check_variable = { process_station_id > 0 }
				meta_trigger = {
					text = { has_global_flag = research_station_[ID]_exists }
					ID = "[?process_station_id]"
				}
			}

			meta_effect = {
				text = {
					set_temp_variable = { station_tier = global.research_station_[ID]_tier }
					set_temp_variable = { station_iteration_progress = global.research_station_[ID]_iteration_progress }
					set_temp_variable = { station_iterations = global.research_station_[ID]_iterations }
					set_temp_variable = { station_controller = global.research_station_[ID]_controller }
					set_temp_variable = { station_fuel_usage = global.research_station_[ID]_output_fuel_usage }
					set_temp_variable = { station_fuel_storage = global.research_station_[ID]_output_storage }
					set_temp_variable = { station_output_power = global.research_station_[ID]_output_power }
					set_temp_variable = { station_active_lab_slots = 0 }
					if = {
						limit = {
							check_variable = { global.research_station_[ID]_module_slot_6 > 0 }
							NOT = { check_variable = { global.research_station_[ID]_module_slot_6_blizzard_damaged > 0 } }
							NOT = { check_variable = { global.research_station_[ID]_module_slot_6_repair_days_remaining > 0 } }
						}
						add_to_temp_variable = { station_active_lab_slots = 1 }
					}
					if = {
						limit = {
							check_variable = { global.research_station_[ID]_module_slot_7 > 0 }
							NOT = { check_variable = { global.research_station_[ID]_module_slot_7_blizzard_damaged > 0 } }
							NOT = { check_variable = { global.research_station_[ID]_module_slot_7_repair_days_remaining > 0 } }
						}
						add_to_temp_variable = { station_active_lab_slots = 1 }
					}
					if = {
						limit = {
							check_variable = { global.research_station_[ID]_module_slot_8 > 0 }
							NOT = { check_variable = { global.research_station_[ID]_module_slot_8_blizzard_damaged > 0 } }
							NOT = { check_variable = { global.research_station_[ID]_module_slot_8_repair_days_remaining > 0 } }
						}
						add_to_temp_variable = { station_active_lab_slots = 1 }
					}
					if = {
						limit = {
							check_variable = { global.research_station_[ID]_module_slot_9 > 0 }
							NOT = { check_variable = { global.research_station_[ID]_module_slot_9_blizzard_damaged > 0 } }
							NOT = { check_variable = { global.research_station_[ID]_module_slot_9_repair_days_remaining > 0 } }
						}
						add_to_temp_variable = { station_active_lab_slots = 1 }
					}
				}
				ID = "[?process_station_id]"
			}

			# Blackout gate for iteration progress:
			# fuel-consuming stations need remaining storage; renewable stations can progress normally.
			if = {
				limit = {
					check_variable = { station_fuel_usage > 0 }
					check_variable = { station_fuel_storage < 1 }
				}
				set_temp_variable = { station_active_lab_slots = 0 }
			}
			if = {
				limit = {
					check_variable = {
						var = station_output_power
						value = 0
						compare = less_than
					}
				}
				set_temp_variable = { station_active_lab_slots = 0 }
			}

			if = {
				limit = { check_variable = { station_tier < 1 } }
				set_temp_variable = { station_tier = 1 }
			}

			# Preserve intended startup tiers for predefined startup stations.
			set_temp_variable = { startup_min_tier = 1 }
			set_temp_variable = { startup_min_iterations = 0 }
			if = {
				limit = { check_variable = { process_station_id < 4 } }
				set_temp_variable = { startup_min_tier = 3 }
				set_temp_variable = { startup_min_iterations = 6 }
			}
			else_if = {
				limit = { check_variable = { process_station_id < 11 } }

				set_temp_variable = { startup_min_iterations = 3 }
			}

			if = {
				limit = { check_variable = { station_iterations < startup_min_iterations } }
				set_temp_variable = { station_iterations = startup_min_iterations }
			}

			set_temp_variable = { station_completed_iteration = 0 }

			if = {
				limit = { check_variable = { station_active_lab_slots > 0 } }
				set_temp_variable = { station_base_iteration_gain = station_active_lab_slots }
				multiply_temp_variable = { station_base_iteration_gain = 0.2 }
				# x7 to compensate for this effect running weekly instead of daily

				add_to_temp_variable = { station_iteration_progress = station_base_iteration_gain }

# Usage
# antarctica_send_supplies_to_selected_station = yes
antarctica_send_supplies_to_selected_station = {
	if = {
		limit = { check_variable = { selected_antarctica_station > 0 } }

		set_temp_variable = { station_id = selected_antarctica_station }
		antarctica_recalculate_station_module_outputs = yes
		set_temp_variable = { delivery_days_remaining = 0 }
		set_temp_variable = { delivery_fuel_pending = 0 }

		meta_effect = {
			text = {
				set_temp_variable = { delivery_days_remaining = global.research_station_[ID]_delivery_days_remaining }
				set_temp_variable = { delivery_fuel_pending = global.research_station_[ID]_delivery_fuel_pending }
			}
			ID = "[?station_id]"
		}

		# Get max capacity and determine delivery parameters
		meta_effect = {
			text = {
				set_temp_variable = { max_capacity = global.research_station_[ID]_output_capacity }
			}
			ID = "[?station_id]"
		}

		if = {
			limit = {
				check_variable = { max_capacity > 0 }
				check_variable = { delivery_days_remaining < 1 }
				check_variable = { delivery_fuel_pending < 1 }
			}

			# Determine delivery time and cost multiplier based on season (April-October = winter = 2x)
			set_temp_variable = { delivery_days = 15 }
			set_temp_variable = { cost_multiplier = 1 }
			if = {
				limit = { OR = { check_variable = { global.month = 4 } check_variable = { global.month = 5 } check_variable = { global.month = 6 } check_variable = { global.month = 7 } check_variable = { global.month = 8 } check_variable = { global.month = 9 } check_variable = { global.month = 10 } } }
				set_temp_variable = { delivery_days = 30 }
				set_temp_variable = { cost_multiplier = 2 }
			}

			# Calculate cost: 0.25b per 1000 fuel = max_capacity * 0.00025 * multiplier
			set_temp_variable = { delivery_cost = max_capacity }
			multiply_temp_variable = { delivery_cost = 0.00025 }
			multiply_temp_variable = { delivery_cost = cost_multiplier }
			round_variable = delivery_cost

			# Deduct cost from treasury using MD's treasury system
			set_temp_variable = { treasury_change = delivery_cost }
			multiply_temp_variable = { treasury_change = -1 }
			modify_treasury_effect = yes

			# Start delivery queue
			meta_effect = {
				text = {
					set_variable = { global.research_station_[ID]_delivery_days_remaining = [DAYS] }
					set_variable = { global.research_station_[ID]_delivery_fuel_pending = [FUEL] }
				}
				ID = "[?station_id]"
				DAYS = "[?delivery_days]"
				FUEL = "[?max_capacity]"
			}

			log = "[GetDateText]: [Root.GetName]: Ordered fuel delivery to Antarctica station [?selected_antarctica_station] for $$[?delivery_cost]b (delivery time: [?delivery_days] days)"

			antarctica_set_selected_station_controller = yes
			add_to_variable = { global.missile_ui_updated = 1 }
		}
		else_if = {
			limit = {
				OR = {
					check_variable = { delivery_days_remaining > 0 }
					check_variable = { delivery_fuel_pending > 0 }
				}
			}
			log = "[GetDateText]: [Root.GetName]: Fuel delivery already in progress for Antarctica station [?selected_antarctica_station]."
		}
	}
}

# Process delivery queue for selected station (decrements counter, adds fuel on completion)
antarctica_process_delivery_queue = {
	for_each_loop = {
		array = global.antarctica_station_ids
		value = process_station_id

		if = {
			limit = {
				check_variable = { process_station_id > 0 }
				meta_trigger = {
					text = {
						has_global_flag = research_station_[ID]_exists
						check_variable = { global.research_station_[ID]_delivery_days_remaining > 0 }
					}
					ID = "[?process_station_id]"
				}
			}

			set_temp_variable = { station_id = process_station_id }
			set_temp_variable = { delivery_complete = 0 }

			meta_effect = {
				text = {
					subtract_from_variable = { global.research_station_[ID]_delivery_days_remaining = 1 }
					if = {
						limit = { check_variable = { global.research_station_[ID]_delivery_days_remaining = 0 } }
						set_temp_variable = { delivery_complete = 1 }
					}
				}
				ID = "[?station_id]"
			}

			if = {
				limit = { check_variable = { delivery_complete = 1 } }

				meta_effect = {
					text = {
						set_temp_variable = { current_storage = global.research_station_[ID]_output_storage }
						set_temp_variable = { pending_fuel = global.research_station_[ID]_delivery_fuel_pending }
						set_temp_variable = { max_capacity = global.research_station_[ID]_output_capacity }
						set_temp_variable = { fuel_usage = global.research_station_[ID]_output_fuel_usage }
					}
					ID = "[?station_id]"
				}

				add_to_temp_variable = { current_storage = pending_fuel }
				if = {
					limit = { check_variable = { current_storage > max_capacity } }
					set_temp_variable = { current_storage = max_capacity }
				}

				set_temp_variable = { fuel_capacity_days = 0 }
				if = {
					limit = { check_variable = { fuel_usage > 0 } }
					set_temp_variable = { fuel_capacity_days = current_storage }
					divide_temp_variable = { fuel_capacity_days = fuel_usage }
					round_variable = fuel_capacity_days
				}

				meta_effect = {
					text = {
						set_variable = { global.research_station_[ID]_output_storage = [STORAGE] }
						set_variable = { global.research_station_[ID]_delivery_fuel_pending = 0 }
						set_variable = { global.research_station_[ID]_output_fuel_capacity_days = [DAYS] }
					}
					ID = "[?station_id]"
					STORAGE = "[?current_storage]"
					DAYS = "[?fuel_capacity_days]"
				}

				log = "[GetDateText]: Fuel delivery completed for Antarctica station [?station_id]. New storage: [?current_storage] fuel ([?fuel_capacity_days] days)"
				add_to_variable = { global.missile_ui_updated = 1 }
			}
		}
	}
}

# Usage
# set_temp_variable = { station_id = id }
# antarctica_apply_station_laboratory_effects_to_controller = yes
antarctica_apply_station_laboratory_effects_to_controller = {
	meta_effect = {
		text = {
			set_temp_variable = { station_controller = global.research_station_[ID]_controller }
			set_temp_variable = { station_fuel_usage = global.research_station_[ID]_output_fuel_usage }
			set_temp_variable = { station_fuel_storage = global.research_station_[ID]_output_storage }
			set_temp_variable = { station_output_power = global.research_station_[ID]_output_power }
			set_temp_variable = { station_illegal_activities = global.research_station_[ID]_illegal_activities_enabled }
		}
		ID = "[?station_id]"
	}

	if = {
		limit = { check_variable = { station_controller = THIS.id } }

		# Fuel blackout gate:
		# - always allow renewable stations (fuel usage <= 0)
		# - for fuel-consuming stations, require remaining storage > 0
		if = {
			limit = {
				OR = {
					check_variable = { station_fuel_usage < 1 }
					check_variable = { station_fuel_storage > 0 }
				}
				check_variable = {
					var = station_output_power
					value = 0
					compare = greater_than_or_equals
				}
			}

			meta_effect = {
				text = {
					for_each_loop = {
						array = global.research_station_[ID]_module_laboratories_list
						value = current_lab_module
						if = {
							limit = { check_variable = { current_lab_module > 0 } }
							set_temp_variable = { station_lab_module = current_lab_module }
							antarctica_apply_single_laboratory_module_effect = yes
						}
					}
				}
				ID = "[?station_id]"
			}
		}
	}
}

antarctica_recalculate_country_laboratory_effects = {
	if = {
		limit = {
			has_idea = at_member
		}

		antarctica_clear_laboratory_country_effects = yes

		if = {
			limit = { has_country_flag = antarctica_has_active_station }

			set_temp_variable = { country_staff_satisfaction_total = 0 }
			set_temp_variable = { country_staff_satisfaction_count = 0 }

			for_each_loop = {
				array = country_owned_station_ids
				value = current_station_id

				if = {
					limit = { check_variable = { current_station_id > 0 } }
					set_temp_variable = { station_id = current_station_id }

					meta_effect = {
						text = {
							set_temp_variable = { station_staff_satisfaction = global.research_station_[ID]_output_staff_satisfaction }
						}
						ID = "[?station_id]"
					}

					add_to_temp_variable = { country_staff_satisfaction_total = station_staff_satisfaction }
					add_to_temp_variable = { country_staff_satisfaction_count = 1 }

					antarctica_apply_station_laboratory_effects_to_controller = yes
				}
			}

			if = {
				limit = { check_variable = { country_staff_satisfaction_count > 0 } }
				set_temp_variable = { country_staff_satisfaction_average = country_staff_satisfaction_total }
				divide_temp_variable = { country_staff_satisfaction_average = country_staff_satisfaction_count }

					add_to_variable = { antarctica_station_total_upkeep_multiplier = global.antarctica_temp_upkeep_cost }
				}
			}
		}
	}
}

# Blizzard System - Triggers blizzard event with 18-35 days duration
# Weekly damage rolls on modules, 14-day auto-repair when blizzard ends
antarctica_trigger_blizzard_event = {
	set_variable_to_random = {
		var = blizzard_duration
		min = 18
		max = 35
		integer = yes
	}

	set_global_flag = antarctica_blizzard_active
	set_variable = { global.antarctica_blizzard_duration_days = blizzard_duration }
	set_variable = { global.antarctica_blizzard_days_remaining = blizzard_duration }
	set_variable = { global.antarctica_blizzard_next_damage_check = 7 }

	log = "[GetDateText]: [Root.GetName]: Antarctica blizzard started. Duration: [?blizzard_duration] days"
}

antarctica_process_blizzard_daily = {
	if = {
		limit = { has_global_flag = antarctica_blizzard_active }
		add_to_variable = { global.antarctica_blizzard_days_remaining = -1 }
		clamp_variable = { var = global.antarctica_blizzard_days_remaining min = 0 }
		add_to_variable = { global.antarctica_blizzard_next_damage_check = -1 }

		if = {
			limit = { check_variable = { global.antarctica_blizzard_next_damage_check < 1 } }
			antarctica_process_blizzard_weekly_damage = yes
			set_variable = { global.antarctica_blizzard_next_damage_check = 7 }
		}

		if = {
			limit = { check_variable = { global.antarctica_blizzard_days_remaining < 1 } }
			clr_global_flag = antarctica_blizzard_active
			antarctica_start_blizzard_repairs = yes
			log = "[GetDateText]: [Root.GetName]: Antarctica blizzard ended. Starting module repairs."
		}
	}
}

antarctica_roll_blizzard_damage_for_slot = {
	if = {
		limit = {
			check_variable = { station_lab_module > 0 }
			meta_trigger = {
				text = {
					NOT = { check_variable = { global.research_station_[ID]_module_slot_[SLOT]_blizzard_damaged > 0 } }
					NOT = { check_variable = { global.research_station_[ID]_module_slot_[SLOT]_repair_days_remaining > 0 } }
				}
				ID = "[?station_id]"
				SLOT = "[?blizzard_slot]"
			}
		}

		set_temp_variable = { out_reliability = 100 }

		if = {
			limit = {
				OR = {
					check_variable = { blizzard_slot = 2 }
					check_variable = { blizzard_slot = 11 }
					check_variable = { blizzard_slot = 13 }
				}
			}
			set_temp_variable = { pwr_module = station_lab_module }
			set_temp_variable = { out_reliability = 0 }
			antarctica_apply_single_powerplant_station_output = yes
		}
		else = {
			set_temp_variable = { lab_module = station_lab_module }
			set_temp_variable = { out_reliability = 0 }
			antarctica_apply_single_laboratory_station_output = yes
		}

		clamp_temp_variable = { var = out_reliability min = 0 max = 100 }
		set_temp_variable_to_random = { var = blizzard_damage_roll min = 0 max = 99 }
		log = "[GetDateText]: [Root.GetName]: Antarctica blizzard check station = [?station_id] slot = [?blizzard_slot] module = [?station_lab_module] reliability = [?out_reliability] roll = [?blizzard_damage_roll]"

		if = {
			limit = {
				check_variable = {
					var = blizzard_damage_roll
					value = out_reliability
					compare = greater_than_or_equals
				}
			}
			antarctica_mark_blizzard_slot_damaged = yes
		}
	}
}
