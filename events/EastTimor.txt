### 2006 East Timor Crisis ###

country_event = {
	id = east_timor.1
	title = east_timor.1.t
	desc = east_timor.1.d
	picture = GFX_timor_resist
	fire_only_once = yes
	is_triggered_only = yes

	# Request international peacekeeping
	option = {
		name = east_timor.1.a
		log = "[GetDateText]: [This.GetName]: east_timor.1.a executed"
		set_global_flag = TIM_2006_crisis
		set_country_flag = TIM_requested_help
		add_stability = -0.10
		add_war_support = 0.05
		hidden_effect = {
			AST = { country_event = { id = east_timor.2 days = 3 } }
			POR = { country_event = { id = east_timor.3 days = 3 } }
			NZL = { country_event = { id = east_timor.4 days = 3 } }
			MAY = { country_event = { id = east_timor.5 days = 3 } }
			news_event = { id = east_timor_news.1 days = 2 }
		}
		ai_chance = { base = 95 }
	}

	# Handle it ourselves
	option = {
		name = east_timor.1.b
		log = "[GetDateText]: [This.GetName]: east_timor.1.b executed"
		set_global_flag = TIM_2006_crisis
		add_stability = -0.15
		add_political_power = -100
		ai_chance = { base = 5 }
	}
}

### News Events ###

news_event = {
	id = east_timor_news.1
	title = east_timor_news.1.t
	desc = east_timor_news.1.desc
	picture = GFX_timor_resist
	major = yes
	is_triggered_only = yes

	option = {
		name = east_timor_news.1.a
		trigger = { original_tag = TIM }
	}

	option = {
		name = east_timor_news.1.b
		trigger = {
			OR = {
				original_tag = AST
				original_tag = POR
				original_tag = NZL
				original_tag = MAY
			}
		}
	}

	option = {
		name = east_timor_news.1.c
		trigger = {
			NOT = { original_tag = TIM }
			NOT = { original_tag = AST }
			NOT = { original_tag = POR }
			NOT = { original_tag = NZL }
			NOT = { original_tag = MAY }
			OR = {
				capital_scope = { is_on_continent = asia }
				capital_scope = { is_on_continent = oceania }
			}
		}
	}

	option = {
		name = east_timor_news.1.d
		trigger = {
			NOT = { original_tag = TIM }
			NOT = { original_tag = AST }
			NOT = { original_tag = POR }
			NOT = { original_tag = NZL }
			NOT = { original_tag = MAY }
			NOT = { capital_scope = { is_on_continent = asia } }
			NOT = { capital_scope = { is_on_continent = oceania } }
		}
	}
}
