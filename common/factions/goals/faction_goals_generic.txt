### Joint Military Exercises

faction_goal_joint_military_exercises = {
	name = joint_military_exercises_goal_name
	description = joint_military_exercises_goal_desc
	category = medium_term
	group = FOCUS_FILTER_ARMY_XP
	visible = {
		has_faction_research_unlocked = yes
	}
	completed = {
		num_faction_members > 5
		count_in_collection = {
			collection = {
				input = game:scope
				operators = { faction_members }
				name = FACTION
			}
			unit = infantry
			equipment_ratio = 0.8
			size > 99
		}
	}
	complete_effect = {
		add_faction_power_projection = 200
		newline = yes
		custom_effect_tooltip = ALL_FACTION_MEMBERS
		every_faction_member = {
			army_experience = 75
			add_tech_bonus = {
				name = joint_exercises_tech_bonus
				bonus = 0.5
				uses = 1
				category = CAT_land_doctrine
			}
		}
	}
	ai_will_do = {
		base = 2
		modifier = {
			add = 3
			has_war = yes
		}
	}
}
