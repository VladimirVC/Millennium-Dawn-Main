on_actions = {
	on_monthly_DEN = {
		effect = {
			if = {
				limit = {
					has_capitulated = no
					NOT = { has_country_flag = DEN_agrar_event_pending }
					NOT = { has_country_flag = DEN_agrar_event_cooldown }
				}
				DEN_Agrar_resolve = yes
			}
			if = {
				limit = {
					has_capitulated = no
					NOT = { has_country_flag = DEN_agrar_event_pending_II }
					NOT = { has_country_flag = DEN_agrar_event_cooldown_II }
				}
				DEN_Agrar_resolve_II = yes
			}
		}
	}
}
