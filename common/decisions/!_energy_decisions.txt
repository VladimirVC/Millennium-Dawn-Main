decision_category_energy = {
	energy_building_battery_parks = {
		icon = GFX_decision_energy_build_button
		allowed = { always = no }
		available = { always = yes }

		days_mission_timeout = ROOT.battery_park_time
		fire_only_once = no
		selectable_mission = yes

		timeout_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision energy_building_battery_parks"
			if = { limit = { has_country_flag = single_battery_park }
				set_temp_variable = { temp_change = 1 }
				build_battery_park_effect = yes
			}
			else = {
				set_temp_variable = { temp_change = 3 }
				build_battery_park_effect = yes
			}
		}
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision energy_building_battery_parks"
			clear_variable = ROOT.battery_park_time
			clr_country_flag = single_battery_park
		}

		ai_will_do = { base = 0 }
	}

	energy_automatic_fuel_purchase = {
		icon = GFX_decision_energy_build_button
		allowed = { always = no }
		available = { always = yes }

		days_mission_timeout = 28
		fire_only_once = no
		selectable_mission = yes

		timeout_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision energy_building_battery_parks"
			buy_fuel_from_the_market_effect = yes
			activate_mission = energy_automatic_fuel_purchase
			set_country_flag = { flag = recently_bought_fuel value = 1 days = 28 }
		}

		ai_will_do = { base = 0 }
	}
}
