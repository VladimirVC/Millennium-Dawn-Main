
# True when the SPA can run at all for this country.
NKO_spa_mechanic_enabled = {
	original_tag = NKO
	has_idea = NKO_spa_umbrella
	NOT = { has_civil_war = yes }
}

# True while fewer than 2 mandates are active — the concurrency cap. Reads the
# live active-mandate counter (NKO_spa_picks_made), which is incremented on
# activation and decremented when a bonus expires. Robust against the variable
# being undefined (pre-init / old save) — undefined treats as 0 active, so the
# player isn't locked out of the mechanic.
NKO_spa_picks_remaining = {
	OR = {
		NOT = { has_variable = NKO_spa_active_mandates }
		check_variable = { NKO_spa_active_mandates < 2 }
	}
}

# Per-decision affordability — only treasury here; PP cost is paid via the
# decision's own `cost` field.
NKO_spa_can_afford_money_03 = { check_variable = { treasury > 0.299 } }
NKO_spa_can_afford_money_04 = { check_variable = { treasury > 0.399 } }
NKO_spa_can_afford_money_05 = { check_variable = { treasury > 0.499 } }
NKO_spa_can_afford_money_07 = { check_variable = { treasury > 0.699 } }
NKO_spa_can_afford_money_075 = { check_variable = { treasury > 0.749 } }
NKO_spa_can_afford_money_15 = { check_variable = { treasury > 1.499 } }

# Research mandate tier markers — representative 1995 / 2005 era techs.
# When reached, the Research Drive bonus is dampened (half / quarter).
NKO_spa_has_1995_tech_tier = {
	has_tech = infantry_weapons_4
	has_tech = nanofibers_3
}

NKO_spa_has_2005_tech_tier = {
	has_tech = infantry_weapons_5
	has_tech = nanofibers_5
}
