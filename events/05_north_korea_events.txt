# Hidden per-host manpower watcher. Seeded by NKO_export_finalize_troops, fires on
# the host every 30 days while our forces are deployed. When the host runs low on
# manpower it appeals to its patron (event_target:NKO_storm_corps_patron).
country_event = {
	id = NKO_export.21
	hidden = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				has_war = yes
				NOT = { has_manpower > 50000 }
				NOT = { has_country_flag = NKO_manpower_request_cooldown }
			}
			set_country_flag = { flag = NKO_manpower_request_cooldown value = 1 days = 120 }
			event_target:NKO_storm_corps_patron = { country_event = { id = NKO_export.20 days = 2 } }
		}
		if = {
			limit = { has_country_flag = NKO_storm_corps_host }
			country_event = { id = NKO_export.21 days = 30 }
		}
	}
}

# A host nation where the Storm Corps is deployed requests manpower
# reinforcements. Fired from the per-host watcher (NKO_export.21) when the host is
# low on manpower. FROM = the requesting host.
country_event = {
	id = NKO_export.20
	title = NKO_export.20.t
	desc = NKO_export.20.d
	picture = GFX_decision_sov_nko_soldiers

	is_triggered_only = yes

	option = {
		name = NKO_export.20.a # Send a tenth of our recruitable manpower
		log = "[GetDateText]: [This.GetName]: NKO_export.20.a executed on [FROM.GetTag]"
		custom_effect_tooltip = NKO_export_send_manpower_tt
		FROM = { set_temp_variable = { NKO_mp_send = manpower } }
		multiply_temp_variable = { NKO_mp_send = 0.1 }
		clamp_temp_variable = { var = NKO_mp_send min = 0 }
		FROM = { add_manpower = NKO_mp_send }
		multiply_temp_variable = { NKO_mp_send = -1 }
		add_manpower = NKO_mp_send
		add_opinion_modifier = { target = FROM modifier = trade_mission }
		FROM = { add_opinion_modifier = { target = NKO modifier = trade_mission } }
		ai_chance = { factor = 70 }
	}

	option = {
		name = NKO_export.20.b # We cannot spare the men
		log = "[GetDateText]: [This.GetName]: NKO_export.20.b executed on [FROM.GetTag]"
		FROM = { add_opinion_modifier = { target = NKO modifier = declined_faction_request } }
		ai_chance = {
			factor = 30
			modifier = { factor = 0  NOT = { has_manpower > 100000 } }
		}
	}
}
