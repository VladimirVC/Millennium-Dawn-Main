# Hong Kong scripted triggers — LegCo motion gating
# Standard policy motion triggers were removed with their motion clocks pending
# the GUI buttons that set the legco_<topic>_motion country flags.

legco_not_has_a_active_motion = {
	NOT = { has_country_flag = HKG_electoral_reform_done }
	NOT = { has_country_flag = HKG_electoral_reform_mainland_a }
	NOT = { has_country_flag = HKG_electoral_reform_mainland_b }
	NOT = { has_country_flag = HKG_electoral_reform_mainland_c }
}

legco_has_a_special_active_motion = {
	OR = {
		has_country_flag = HKG_electoral_reform_done
		has_country_flag = HKG_electoral_reform_mainland_a
		has_country_flag = HKG_electoral_reform_mainland_b
		has_country_flag = HKG_electoral_reform_mainland_c
	}
}
