# Function: state_owned_or_puppet_owned_by_root
# Purpose: State-scoped trigger. Returns true if the state is owned by ROOT or by a subject of ROOT.
# Used inside count_triggers blocks in *_integrate_start available checks so puppet-held cores count toward the 80% threshold.
state_owned_or_puppet_owned_by_root = {
	OR = {
		is_owned_by = ROOT
		OWNER = { is_subject_of = ROOT }
	}
}

# Function: formable_nation_check
# Purpose: The check allows us to simplify and streamline the beahvior of the code down to one specific catch all
formable_nation_check = {
	OR = {
		var:tag_check = {
			is_subject_of = ROOT
			has_war = no
		}
		AND = {
			var:tag_check = {
				all_core_state = {
					OR = {
						is_owned_and_controlled_by = ROOT
						is_core_of = ROOT
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = general_integrate_tt
				var:tag_check = {
					ROOT = {
						check_variable = { core_compliance@PREV > 50 }
						check_variable = { core_resistance@PREV < 30 }
					}
				}
			}
		}
	}
}
