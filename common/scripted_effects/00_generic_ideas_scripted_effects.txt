upgrade_western_boost_idea = {
	if = {
		limit = {
			NOT = { has_idea = generic_western_boost_idea }
			NOT = { has_idea = generic_western_boost_idea2 }
		}
		add_timed_idea = {
			idea = generic_western_boost_idea
			days = 365
		}
	}
	else_if = {
		limit = { has_idea = generic_western_boost_idea }
		remove_ideas = generic_western_boost_idea
		add_timed_idea = {
			idea = generic_western_boost_idea2
			days = 365
		}
	}

}

improve_civilian_industrial_projects = {
	if = {
		limit = { NOT = { has_idea = civilian_industry_project } }
		add_ideas = civilian_industry_project
	}
	else_if = {
		limit = { has_idea = civilian_industry_project }
		swap_ideas = {
			remove_idea = civilian_industry_project
			add_idea = civilian_industry_project2
		}
	}
	else_if = {
		limit = { has_idea = civilian_industry_project2 }
		swap_ideas = {
			remove_idea = civilian_industry_project2
			add_idea = civilian_industry_project3
		}
	}
}
