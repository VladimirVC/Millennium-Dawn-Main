# Faction theaters for Millennium Dawn's modern map.
# Region IDs reference map/strategicregions/ (1-220). Each theater groups
# contiguous land regions plus the sea regions that bridge them, so the area
# stays connected (engine requirement).
#
# Following vanilla conventions:
#  - can_skip_first_region = yes  -> theater can form even if the first-listed
#    region is not available to the faction leader (these theaters are large).
#  - cancel = { has_war = no }    -> theaters are wartime constructs; abandon at peace.
#  - ai_will_do has a factor = 0 / has_war = no guard so theaters are not created
#    in peacetime. Weight is otherwise added for leaders whose alliance or wars
#    make the theater relevant.

# # EUROPE

western_europe = {
	name = theater_western_europe
	can_skip_first_region = yes
	cancel = {
		has_war = no
	}
	ai_will_do = {
		base = 0
		modifier = {
			add = 1
			capital_scope = { is_on_continent = europe }
		}
		modifier = {
			add = 5
			OR = {
				original_tag = USA
				is_in_faction_with = USA
				has_idea = NATO_member
				has_idea = EU_member
			}
		}
		modifier = {
			add = 5

				has_war_with = FRA
