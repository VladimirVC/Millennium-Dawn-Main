### Voting Decision triggers ####
#################################

is_socialist_block = {
	OR = {
		western_social_democrats_are_in_power = yes
		neutrality_neutral_social_are_in_power = yes
	}
}

EU_voting_decision_result_trigger = {
	check_variable = { global.EU_pop_ratio_yes > 0.65 }
	set_temp_variable = { total_eu_members = global.EU_active_vote_yes }
	add_to_temp_variable = { total_eu_members = global.EU_active_vote_no }
	set_temp_variable = { vote_numbers = global.EU_active_vote_yes }
	divide_temp_variable = { vote_numbers = total_eu_members }
	check_variable = { vote_numbers > 0.55 }
}

### EU_mff ###

focus_EU_mff_draft = {

		array = global.EU_member
		has_country_flag = EU_mff_draft
	}
}
