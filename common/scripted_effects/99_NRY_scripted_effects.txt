# Norwegian event helper effects

# Leopard 2A5 purchase from Germany; reads no inputs, buyer is always NRY regardless of ROOT
NRY_GER_tank_purchase = {
	if = {
		limit = { has_dlc = "No Step Back" }
		NRY = {
			set_temp_variable = { treasury_change = -45.29 }
			modify_treasury_effect = yes
			add_equipment_to_stockpile = {
				type = medium_tank_chassis_1
				variant_name = "Leopard 2A5"
				amount = 45
				producer = GER
			}
		}
		if = {
			limit = { country_exists = GER }
			GER = {
				set_temp_variable = { treasury_change = 45.29 }
				modify_treasury_effect = yes
				add_equipment_to_stockpile = {
					type = medium_tank_chassis_1
					variant_name = "Leopard 2A5"
					amount = -45
					producer = GER
				}
			}
		}
	}
	else = {
		NRY = {
			set_temp_variable = { treasury_change = -45.29 }
			modify_treasury_effect = yes
			add_equipment_to_stockpile = {
				type = MBT_2
				amount = 45
				producer = GER
			}
		}
		if = {
			limit = { country_exists = GER }
			GER = {
				set_temp_variable = { treasury_change = 45.29 }
				modify_treasury_effect = yes
				add_equipment_to_stockpile = {
					type = MBT_2
					amount = -45
					producer = GER
				}
			}
		}
	}
}
