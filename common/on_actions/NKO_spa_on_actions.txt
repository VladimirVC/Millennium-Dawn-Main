
on_actions = {

	# -------------------------------------------------------------------------
	# Game start — intentionally NO automatic SPA install. The mechanic is
	# unlocked when the player completes the NKO_humanitarian_aid focus
	# (see common/national_focus/korea_north.txt). The focus completion_reward
	# calls NKO_spa_initialise + fires the inaugural event NKO_spa.0.
	# -------------------------------------------------------------------------

	# -------------------------------------------------------------------------
	# Civil-war start — strip SPA state from both sides immediately.
	# Direct effect call (no event hop) for the lowest possible overhead.
	# -------------------------------------------------------------------------
	on_civil_war_started = {
		effect = {
			if = {
				limit = { original_tag = NKO }
				NKO_spa_suppress_for_collapse = yes
			}
			if = {
				limit = { FROM = { original_tag = NKO } }
				FROM = { NKO_spa_suppress_for_collapse = yes }
			}
		}
	}
}
