# Greek Creditor System Triggers

GRE_is_eligible_creditor = {
	custom_trigger_tooltip = {
		tooltip = GRE_eligible_creditor_tt
		NOT = { has_war_with = GRE }
		is_subject = no
		has_opinion = { target = GRE value > -50 }
		NOT = { is_in_array = { GRE.GRE_creditors = THIS.id } }
	}
}

GRE_has_creditors = {
	check_variable = { GRE_creditors^num > 0 }
}

GRE_is_creditor_of_greece = {
	is_in_array = { GRE.GRE_creditors = THIS.id }
}

GRE_austerity_ongoing = {
	OR = {
		has_idea = GRE_austerity_1
		has_idea = GRE_austerity_2
		has_idea = GRE_austerity_3
	}
}
