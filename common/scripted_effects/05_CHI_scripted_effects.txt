CHI_bri_request_project = {
	set_country_flag = bri_pipeline_busy
	CHI = {
		add_to_variable = { CHI_bri_committed_projects = 1 }
		subtract_from_variable = { var = CHI_bri_treasury value = CHI_bri_project_cost }
		clamp_variable = { var = CHI_bri_treasury min = 0 }
	}

	if = {
		limit = { CHI = { has_country_flag = CHI_bri_auto_accept } }
		CHI_bri_start_project = yes
	}
	else = {
		set_country_flag = bri_pending_application
		CHI = { country_event = { id = chi_bri.10 days = 3 } }
	}
}
CHI_bri_start_project = {
	if = {
		limit = {
			has_country_flag = bri_pipeline_busy
			country_exists = CHI
		}
		clr_country_flag = bri_pending_application
		set_variable = { bri_project_granted = 0 }

		set_temp_variable = { bri_cost = CHI.CHI_bri_project_cost }

		if = {
			limit = { check_variable = { treasury > bri_cost } }
			set_temp_variable = { treasury_change = bri_cost }
			multiply_temp_variable = { treasury_change = -1 }
			modify_treasury_effect = yes
			CHI = {
				set_temp_variable = { treasury_change = PREV.bri_cost }
				modify_treasury_effect = yes
			}
		}
		else = {
			CHI_bri_issue_loan = yes
			country_event = { id = chi_bri.23 days = 1 }
		}

		activate_mission = CHI_bri_project_mission
	}
}

CHI_bri_complete_project = {
	if = {
		limit = { has_country_flag = bri_pipeline_busy }
		clr_country_flag = bri_pipeline_busy
		if = {
			limit = { country_exists = CHI }
			CHI = {
				subtract_from_variable = { CHI_bri_committed_projects = 1 }
				clamp_variable = { var = CHI_bri_committed_projects min = 0 }
			}
		}

		CHI_bri_grant_project_reward = yes

		if = {
			limit = {
				check_variable = { bri_project_granted = 1 }
				country_exists = CHI
			}
			add_to_variable = { bri_projects_completed = 1 }
			CHI_bri_advance_named_project = yes
			CHI = { add_to_variable = { CHI_bri_completed_projects = 1 } }
			CHI_bri_apply_partner_rank = yes

			CHI = {
				set_temp_variable = { int_investment_change = CHI_bri_return_per_project }
				set_temp_variable = { bri_return_mult = 1 }
				add_to_temp_variable = { bri_return_mult = CHI_bri_returns_multiplier_base }
				add_to_temp_variable = { bri_return_mult = modifier@CHI_bri_returns_multiplier }
				multiply_temp_variable = { int_investment_change = bri_return_mult }
				modify_international_investment_effect = yes
			}
		}
	}
}

# Applies the demographic dividend and population penalty for the current one-child
# policy stage (CHI_demographic_policy_stage 1-4). monthly_population is exclusive to
# this system; the other three factors are shared with focuses, so the previously
# applied stage is reversed before the current one is added.
CHI_apply_demographic_policy = {
	if = {
		limit = { check_variable = { CHI_demographic_stage_applied = 1 } }
		add_to_variable = { CHI_economy_monthly_population = 0.90 }
		subtract_from_variable = { CHI_economy_productivity_growth_modifier = 0.08 }
		add_to_variable = { CHI_economy_consumer_goods_factor = 0.10 }
		subtract_from_variable = { CHI_economy_total_workforce_modifier = 0.12 }
	}
	else_if = {
		limit = { check_variable = { CHI_demographic_stage_applied = 2 } }
		add_to_variable = { CHI_economy_monthly_population = 0.45 }
		subtract_from_variable = { CHI_economy_productivity_growth_modifier = 0.04 }
		add_to_variable = { CHI_economy_consumer_goods_factor = 0.05 }
		subtract_from_variable = { CHI_economy_total_workforce_modifier = 0.06 }
	}
	else_if = {
		limit = { check_variable = { CHI_demographic_stage_applied = 3 } }
		add_to_variable = { CHI_economy_monthly_population = 0.15 }
		subtract_from_variable = { CHI_economy_productivity_growth_modifier = 0.015 }
		add_to_variable = { CHI_economy_consumer_goods_factor = 0.02 }
		subtract_from_variable = { CHI_economy_total_workforce_modifier = 0.02 }
	}

	custom_effect_tooltip = { localization_key = modifies_dynamic_modifier_tt MODIFIER = CHI_economy_modifier }
	if = {
		limit = { check_variable = { CHI_demographic_policy_stage = 1 } }
		subtract_from_variable = { CHI_economy_monthly_population = 0.90 tooltip = monthly_population_tt }
		add_to_variable = { CHI_economy_productivity_growth_modifier = 0.08 tooltip = productivity_growth_modifier_tt }
		subtract_from_variable = { CHI_economy_consumer_goods_factor = 0.10 tooltip = consumer_goods_factor_tt }
		add_to_variable = { CHI_economy_total_workforce_modifier = 0.12 tooltip = total_workforce_modifier_tt }
	}
	else_if = {
		limit = { check_variable = { CHI_demographic_policy_stage = 2 } }
		subtract_from_variable = { CHI_economy_monthly_population = 0.45 tooltip = monthly_population_tt }
		add_to_variable = { CHI_economy_productivity_growth_modifier = 0.04 tooltip = productivity_growth_modifier_tt }
		subtract_from_variable = { CHI_economy_consumer_goods_factor = 0.05 tooltip = consumer_goods_factor_tt }
		add_to_variable = { CHI_economy_total_workforce_modifier = 0.06 tooltip = total_workforce_modifier_tt }
	}
	else_if = {
		limit = { check_variable = { CHI_demographic_policy_stage = 3 } }
		subtract_from_variable = { CHI_economy_monthly_population = 0.15 tooltip = monthly_population_tt }
		add_to_variable = { CHI_economy_productivity_growth_modifier = 0.015 tooltip = productivity_growth_modifier_tt }
		subtract_from_variable = { CHI_economy_consumer_goods_factor = 0.02 tooltip = consumer_goods_factor_tt }
		add_to_variable = { CHI_economy_total_workforce_modifier = 0.02 tooltip = total_workforce_modifier_tt }
	}
	# stage 4 (abolished): penalty and dividend both fully reversed above, nothing applied

	set_variable = { CHI_demographic_stage_applied = CHI_demographic_policy_stage }
}

### Russian arms dependency: re-evaluate tier from CHI_military_self_sufficiency. Idempotent, downgrade-only. ###

CHI_bri_issue_loan = {
	set_temp_variable = { bri_principal = CHI.CHI_bri_project_cost }
	set_variable = { bri_loan_rate = 4 }
	add_to_variable = { var = bri_debt_owed_to_china value = bri_principal }

	set_variable = { bri_loan_payment = bri_debt_owed_to_china }
	multiply_variable = { bri_loan_payment = 0.00019 }
	multiply_variable = { bri_loan_payment = bri_loan_rate }

	CHI = {
		set_temp_variable = { bri_debtor_id = PREV.id }
		set_temp_variable = { bri_loan_amount = PREV.bri_principal }

		if = {
			limit = { is_in_array = { CHI_bri_debtors = bri_debtor_id } }
			CHI_bri_find_debtor_index = yes
			if = {
				limit = { check_variable = { bri_found_index > -1 } }
				add_to_variable = { var = CHI_bri_loan_principal^bri_found_index value = bri_loan_amount }
			}
		}
		else = {
			add_to_array = { CHI_bri_debtors = bri_debtor_id }
			set_temp_variable = { bri_new_index = CHI_bri_debtors^num }
			subtract_from_temp_variable = { bri_new_index = 1 }
			set_variable = { CHI_bri_loan_principal^bri_new_index = bri_loan_amount }
			set_variable = { CHI_bri_loan_rate^bri_new_index = 4 }
		}
	}
}
CHI_bri_process_loans = {
	set_variable = { CHI_bri_loan_income = 0 }

	if = {
		limit = { check_variable = { CHI_bri_debtors^num > 0 } }
		for_each_loop = {
			array = CHI_bri_debtors
			value = v
			index = i

			set_temp_variable = { bri_interest = CHI_bri_loan_principal^i }
			multiply_temp_variable = { bri_interest = 0.00019 }
			multiply_temp_variable = { bri_interest = CHI_bri_loan_rate^i }
			add_to_variable = { var = CHI_bri_loan_income value = bri_interest }

			set_temp_variable = { bri_amortisation = CHI_bri_loan_principal^i }
			multiply_temp_variable = { bri_amortisation = 0.05 }

			var:CHI_bri_debtors^i = {
				set_variable = { bri_loan_payment = ROOT.bri_interest }
				if = {
					limit = {
						check_variable = { treasury > ROOT.bri_amortisation }
						NOT = { has_active_mission = bankruptcy_incoming_collapse }
					}
					set_temp_variable = { treasury_change = ROOT.bri_amortisation }
					multiply_temp_variable = { treasury_change = -1 }
					modify_treasury_effect = yes
					subtract_from_variable = { var = bri_debt_owed_to_china value = ROOT.bri_amortisation }
					clamp_variable = { var = bri_debt_owed_to_china min = 0 }
					clr_country_flag = bri_defaulted
				}
				else_if = {
					limit = { NOT = { has_country_flag = bri_defaulted } }
					set_country_flag = bri_defaulted
					country_event = { id = chi_bri.16 days = 1 }
					ROOT = {
						add_to_variable = { CHI_bri_defaulted_projects = 1 }
						country_event = { id = chi_bri.19 days = 1 }
					}
				}
			}

			if = {
				limit = { var:CHI_bri_debtors^i = { NOT = { has_country_flag = bri_defaulted } } }
				subtract_from_variable = { var = CHI_bri_loan_principal^i value = bri_amortisation }
				set_temp_variable = { treasury_change = bri_amortisation }
				modify_treasury_effect = yes
			}
		}

		CHI_bri_cleanup_loans = yes
	}
}
CHI_bri_cleanup_loans = {
	set_temp_variable = { bri_cleanup_index = CHI_bri_debtors^num }
	subtract_from_temp_variable = { bri_cleanup_index = 1 }

	while_loop_effect = {
		limit = { check_variable = { bri_cleanup_index > -1 } }

		if = {
			limit = { check_variable = { CHI_bri_loan_principal^bri_cleanup_index < 0.01 } }
			var:CHI_bri_debtors^bri_cleanup_index = {
				set_variable = { bri_debt_owed_to_china = 0 }
				set_variable = { bri_loan_payment = 0 }
				set_variable = { bri_loan_rate = 0 }
				clr_country_flag = bri_defaulted
			}
			remove_from_array = { array = CHI_bri_debtors index = bri_cleanup_index }
			remove_from_array = { array = CHI_bri_loan_principal index = bri_cleanup_index }
			remove_from_array = { array = CHI_bri_loan_rate index = bri_cleanup_index }
		}

		subtract_from_temp_variable = { bri_cleanup_index = 1 }
	}

	if = {
		limit = { check_variable = { CHI_bri_debtors^num = 0 } }
		set_variable = { CHI_bri_loan_income = 0 }
	}
}
