## COVID Events
# A New Virus
country_event = {
	id = COVID_events.0
	title = COVID_events.0.t
	desc = COVID_events.0.d
	picture = GFX_african_school # FIXME: Replace with proper pandemic art
	is_triggered_only = yes

	option = {
		name = COVID_events.0.a
		log = "[GetDateText]: [This.GetName]: COVID_events.0.a executed"
		set_global_flag = COVID_start_of_pandemic

		hidden_effect = {
			set_variable = { global.COVID_start_nation = THIS.id }
			country_event = { id = COVID_events.1 days = 15 random_days = 15 }

			# Global pandemic spread — fires to every country ~3 months after initial outbreak
			every_country = {
				limit = {
					has_capitulated = no
				}
				country_event = { id = COVID_events.4 days = 90 random_days = 90 }
				country_event = { id = COVID_events.5 days = 180 random_days = 60 }
			}

			# Germany pandemic flavor event
			if = {
				limit = { country_exists = GER }
				GER = {
					country_event = { id = germany_yearly.49 days = 120 random_days = 60 }
				}
			}
		}

		ai_chance = {
			base = 1
		}
	}
}

		add_timed_idea = {

			if = { limit = { country_exists = OMA } OMA = { country_event = { id = pandemic_events.7 days = 30 random_days = 30 } } }
