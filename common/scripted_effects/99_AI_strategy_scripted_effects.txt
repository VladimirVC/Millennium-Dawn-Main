ai_update_build_units = { #default is not to build units
	if = { # Set flag — allow unit building
		limit = {
			NOT = { has_country_flag = AI_is_threatened }
			OR = {
				has_war = yes
				is_subject = yes
				AND = {
					has_government = democratic
					threat > 0.15
				}
				AND = {
					has_government = neutrality
					threat > 0.30
				}
				AND = {
					NOT = { has_government = democratic }
					NOT = { has_government = neutrality }
					threat > 0.10
				}
				has_government = nationalist
				has_government = fascism
				check_variable = { potential_and_current_enemies^num > 0 }
			}
		}
		set_country_flag = AI_is_threatened
		add_to_variable = { var = ai_wants_divisions value = 100 }
		log = "[GetDateText]: [THIS.GetName]: AI Update - AI Build Unit Flag Set"
	}
	if = { # Clear flag — stop building
		limit = {
			has_country_flag = AI_is_threatened
			is_subject = no
			has_war = no
			NOT = { has_government = nationalist }
			NOT = { has_government = fascism }
			OR = {
				AND = {
					has_government = democratic
					threat < 0.15
				}
				AND = {
					has_government = neutrality
					threat < 0.30
				}
				AND = {
					NOT = { has_government = democratic }
					NOT = { has_government = neutrality }
					threat < 0.10
				}
			}
			check_variable = { potential_and_current_enemies^num < 1 }
		}
		clr_country_flag = AI_is_threatened
		log = "[GetDateText]: [THIS.GetName]: AI Update - AI Build Unit Flag clear"
		add_to_variable = { var = ai_wants_divisions value = -100 }
	}
}
