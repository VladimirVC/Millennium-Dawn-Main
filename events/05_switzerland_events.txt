			base = 0
			modifier = {
				add = 200
				is_subject_of = FROM
			}
			modifier = {
				add = 50
				has_capitulated = yes
			}
			modifier = {
				add = 25
				surrender_progress > 0.3
			}
			modifier = {
				add = 15
				same_ideology = yes
				has_opinion = {
					target = FROM
					value > 80
				}
			}
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		if = {
			limit = {
				FROM = { exists = yes }
				NOT = { has_war_with = FROM }
			}
			if = {
				limit = { owns_state = 857 }
				FROM = { transfer_state = 857 } # Savoy
			}
			if = {
				limit = { owns_state = 63 }
				FROM = { transfer_state = 63 } # Provence
			}
			if = {
				limit = { owns_state = 1000 }
				FROM = { transfer_state = 1000 } # Nice
			}
			FROM = {
				country_event = { id = switzerland.9 days = 1 }
				add_opinion_modifier = {
					target = ROOT
					modifier = gratitude
				}
			}
		}
		else_if = {
			limit = { FROM = { exists = yes } }
			add_opinion_modifier = {
				target = FROM
				modifier = small_decrease
			}
		}
	}

	option = {
		name = switzerland.8.b
		log = "[GetDateText]: [This.GetName]: switzerland.8.b executed"
		ai_chance = {
			base = 100
			modifier = {
				factor = 0.5
				sender_influence_higher_30 = yes
			}
		}
		FROM = {
			country_event = { id = switzerland.10 days = 1 }
		}
		add_opinion_modifier = {
			target = FROM
			modifier = insult
		}
	}

}
