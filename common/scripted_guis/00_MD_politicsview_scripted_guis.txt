						array = parties_list_sorted
						size = pos_array^num
					}

					for_each_loop = {
						array = pos_array

						remove_from_array = {
							array = parties_list_sorted
							index = v
						}

						add_to_array = {
							array = parties_list_sorted
							index = v
							value = v
						}

						if = {
							limit = {
								is_party_in_government = no
								check_variable = { party_pop_array^v > 0.009 }
							}

							add_to_array = {
								array = opposition_array
								value = v
							}

						}
					}

					remove_from_array = {
						array = opposition_array
						index = 0
					}

					update_dirty_politicsview_expanded = yes
				}
				else = {
					clr_country_flag = MD_politicsview_expanded_open
				}
			}
		}
