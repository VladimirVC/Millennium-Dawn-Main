BLR_can_send_azot_mission = {
	custom_trigger_tooltip = {
		tooltip = BLR_can_send_azot_mission_TT
		check_variable = { ROOT.current_azot_capacity < ROOT.current_max_azot_capacity }
	}
}

		NOT = { has_decision = BLR_mission_to_France }
		NOT = { has_decision = BLR_mission_to_Norway }
		NOT = { has_decision = BLR_mission_to_Sweden }
		NOT = { has_decision = BLR_mission_to_Russia }
		NOT = { has_decision = BLR_mission_to_Ukraine }
		NOT = { has_decision = BLR_mission_to_Kazahstan }
		NOT = { has_decision = BLR_mission_to_Czechia }
		NOT = { has_decision = BLR_mission_to_Netherland }
		NOT = { has_country_flag = BLR_pending_offer }
	}
}
