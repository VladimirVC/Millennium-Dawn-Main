### Chinese Trigger Conditions

is_sco = {
	OR = {
		has_idea = CHI_sco_member
		has_idea = CHI_sco_member_econ
		has_idea = CHI_sco_member_pol
		has_idea = CHI_sco_member_mil
	}
}

CHI_sco_is_dialogue_partner = {
	has_idea = CHI_sco_dialogue_partner
}

CHI_sco_is_observer = {
	has_idea = CHI_sco_observer
}

CHI_sco_is_any = {
	OR = {
		has_idea = CHI_sco_dialogue_partner
		has_idea = CHI_sco_observer
		has_idea = CHI_sco_member
		has_idea = CHI_sco_member_econ
		has_idea = CHI_sco_member_pol
		has_idea = CHI_sco_member_mil
	}
}

CHI_sco_can_apply_to_join = {
	CHI = { has_completed_focus = CHI_Form_the_SCO }
	NOT = { CHI_sco_is_any = yes }
	NOT = { is_in_faction_with = USA }
	NOT = { is_in_faction_with = ENG }
	CHI = { has_war = no }
	NOT = { has_war_with = CHI }
}

all_sco_members_are_china_ruling_party = {
	if = {
		limit = { has_government = communism }
		custom_trigger_tooltip = {
			tooltip = SCO_all_emerging_tt
			NOT = {
				any_of_scopes = {
					array = global.sco_members
					is_sco = yes
					NOT = { has_government = communism }
				}
			}
		}
	}
	else_if = {
		limit = { has_government = democratic }
		custom_trigger_tooltip = {
			tooltip = SCO_all_democratic_tt
			NOT = {
				any_of_scopes = {
					array = global.sco_members
					is_sco = yes
					NOT = { has_government = democratic }
				}
			}
		}
	}
	else = { always = no }
}

CHI_sco_observer_eligible_special = {
	OR = {
		AND = { #Italy
			original_tag = ITA
			is_subject = no
			has_completed_focus = ITA_attempt_join_sco
			CHI = { has_country_flag = accepted_italy_as_sco_candidate }
			CHI_sco_can_apply_observer = yes
		}
		AND = { #Sweden
			original_tag = SWE
			has_completed_focus = SWE_SCO_observer_status
			CHI_sco_can_apply_observer = yes
		}
		AND = { #Ukraine
			original_tag = UKR
			is_subject = no
			CHI = { has_completed_focus = CHI_Form_the_SCO }
			OR = {
				AND = {
					emerging_communist_state_are_in_power = yes
					has_completed_focus = UKR_commie_china_trade
				}
				AND = {
					emerging_reactionaries_are_in_power = yes
					has_completed_focus = UKR_party_regions_inaguration
				}
			}
			CHI_sco_can_apply_observer = yes
		}
		AND = { #Libya
			original_tag = LBA
			is_subject = no
			CHI = {
				has_completed_focus = CHI_Form_the_SCO
				has_completed_focus = CHI_China_Africa_Development_Fund
			}
			CHI_sco_can_apply_observer = yes
			LBA = {
				OR = {
					NOT = {
						OR = {
							has_completed_focus = LBA_strive_for_african_union

						}
					}
					NOT = { salafist_caliphate_are_in_power = yes }
				}
			}
		}
	}
}
