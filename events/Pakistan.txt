### Tribal Areas Pressure Chain ###

# American pressure on the tribal areas
country_event = {
	id = pakistan.1
	title = pakistan.1.t
	desc = pakistan.1.d
	picture = GFX_afghanistan_war
	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		country_exists = TTP
	}

	# Launch military operation in Waziristan
	option = {
		name = pakistan.1.a
		log = "[GetDateText]: [This.GetName]: pakistan.1.a executed"
		add_war_support = 0.05
		add_stability = -0.05
		add_opinion_modifier = { target = USA modifier = medium_increase }
		hidden_effect = {
			USA = {
				country_event = { id = pakistan.2 days = 3 }
				add_war_support = 0.03
			}
			country_event = { id = pakistan.4 days = 120 }
			news_event = { id = pakistan_news.1 days = 1 }
		}
		ai_chance = { base = 70 }
	}

	# Negotiate with tribal elders
	option = {
		name = pakistan.1.b
		log = "[GetDateText]: [This.GetName]: pakistan.1.b executed"
		add_stability = 0.03
		add_political_power = 25
		hidden_effect = {
			USA = { country_event = { id = pakistan.3 days = 7 } }
		}
		ai_chance = { base = 20 }
	}

	# Refuse American demands
	option = {
		name = pakistan.1.c
		log = "[GetDateText]: [This.GetName]: pakistan.1.c executed"
		reverse_add_opinion_modifier = { target = USA modifier = large_decrease }
		add_political_power = -50
		hidden_effect = {
			USA = { country_event = { id = pakistan.3 days = 7 } }
		}
		ai_chance = { base = 10 }
	}
}
