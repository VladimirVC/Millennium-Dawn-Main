# -----------------------------------------------------------------------------
# Initialise — adds the umbrella idea + 8 standing debuffs and seeds the
# picks counter. Idempotent (each idea add is gated). Called from on_startup.
# -----------------------------------------------------------------------------
NKO_spa_initialise = {
	hidden_effect = {
		if = {
			limit = { NOT = { has_idea = NKO_spa_umbrella } }
			add_ideas = NKO_spa_umbrella
		}

		if = {
			limit = {
				NOT = { has_idea = NKO_spa_debuff_industry }
				NOT = { has_idea = NKO_spa_bonus_industry }
			}
			add_ideas = NKO_spa_debuff_industry
		}
		if = {
			limit = {
				NOT = { has_idea = NKO_spa_debuff_production }
				NOT = { has_idea = NKO_spa_bonus_production }
			}
			add_ideas = NKO_spa_debuff_production
		}
		if = {
			limit = {
				NOT = { has_idea = NKO_spa_debuff_military }
				NOT = { has_idea = NKO_spa_bonus_military }
			}
			add_ideas = NKO_spa_debuff_military
		}
		if = {
			limit = {
				NOT = { has_idea = NKO_spa_debuff_economy }
				NOT = { has_idea = NKO_spa_bonus_economy }
			}
			add_ideas = NKO_spa_debuff_economy
		}
		if = {
			limit = {
				NOT = { has_idea = NKO_spa_debuff_research }
			}
			add_ideas = NKO_spa_debuff_research
		}
		if = {
			limit = {
				NOT = { has_idea = NKO_spa_debuff_agriculture }
				NOT = { has_idea = NKO_spa_bonus_agriculture }
			}
			add_ideas = NKO_spa_debuff_agriculture
		}
		if = {
			limit = {
				NOT = { has_idea = NKO_spa_debuff_foreign }
				NOT = { has_idea = NKO_spa_bonus_foreign }
			}
			add_ideas = NKO_spa_debuff_foreign
		}
		if = {
			limit = {
				NOT = { has_idea = NKO_spa_debuff_construction }
				NOT = { has_idea = NKO_spa_bonus_construction }
			}
			add_ideas = NKO_spa_debuff_construction
		}

		# Seed the picks counter at 0 so the .gui's [?picks_made|0] token has
		# a real value to render on the very first frame.
		set_variable = { NKO_spa_picks_made = 0 }
	}
}

NKO_arms_export_initialize = {
	set_country_flag = NKO_arms_export_active
	if = { limit = { NOT = { has_variable = NKO_arms_export_clients } }  set_variable = { NKO_arms_export_clients = 0 } }
	if = { limit = { NOT = { has_variable = NKO_arms_export_deals } }    set_variable = { NKO_arms_export_deals = 0 } }
	if = { limit = { NOT = { has_variable = NKO_arms_export_deals_since_west } } set_variable = { NKO_arms_export_deals_since_west = 0 } }
	if = { limit = { NOT = { has_variable = NKO_arms_export_heat } }     set_variable = { NKO_arms_export_heat = 0 } }
	if = { limit = { NOT = { has_variable = NKO_arms_export_revenue } }  set_variable = { NKO_arms_export_revenue = 0 } }
	if = { limit = { NOT = { has_variable = NKO_arms_export_output_var } } set_variable = { NKO_arms_export_output_var = 0 } }
	if = { limit = { NOT = { has_variable = NKO_arms_export_gui_dirty } } set_variable = { NKO_arms_export_gui_dirty = 0 } }
	if = {
		limit = { NOT = { has_dynamic_modifier = { modifier = NKO_arms_export_modifiers } } }
		add_dynamic_modifier = { modifier = NKO_arms_export_modifiers }
	}
	NKO_arms_export_rebuild_targets = yes
	NKO_arms_contract_initialize = yes
}

NKO_arms_export_rebuild_targets = {
	clear_array = global.NKO_arms_export_targets
	if = {
		limit = { has_country_flag = NKO_arms_export_conv_open }
		every_country = {
			limit = {
				exists = yes
				has_war = yes
				NOT = { original_tag = NKO }
				is_subject = no
				NOT = { has_war_with = NKO }
			}
			add_to_array = { global.NKO_arms_export_targets = THIS }
		}
	}
}

NKO_rebuild_patrons = {
	clear_array = global.NKO_patrons_at_war
	if = {
		limit = {
			country_exists = SOV
			NOT = { has_war_with = SOV }
			SOV = { has_war = yes }
		}
		add_to_array = { global.NKO_patrons_at_war = SOV.id }
	}
	if = {
		limit = {
			country_exists = CHI
			NOT = { has_war_with = CHI }
			CHI = { has_war = yes }
		}
		add_to_array = { global.NKO_patrons_at_war = CHI.id }
	}
}

	}
}
