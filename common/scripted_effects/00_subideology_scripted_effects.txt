# steal_from_party: helper for steal_popularity_from_* effects
# Parameters: steal_party_index (0-23)
# Steals 1% popularity from the given party and gives it to the ruling party
steal_from_party = {
	if = {
		limit = { check_variable = { party_pop_array^steal_party_index > 0.01 } }
		hidden_effect = {
			if = {
				limit = {
					check_variable = { steal_party_index < 4 }
					NOT = { has_government = democratic }
				}
				add_popularity = { ideology = democratic popularity = -0.01 }
			}
			else_if = {
				limit = {
					check_variable = { steal_party_index > 3 }
					check_variable = { steal_party_index < 10 }
					NOT = { has_government = communism }
				}
				add_popularity = { ideology = communism popularity = -0.01 }
			}
			else_if = {
				limit = {
					check_variable = { steal_party_index > 9 }
					check_variable = { steal_party_index < 12 }
					NOT = { has_government = fascism }
				}
				add_popularity = { ideology = fascism popularity = -0.01 }
			}
			else_if = {
				limit = {
					check_variable = { steal_party_index > 11 }
					check_variable = { steal_party_index < 20 }
					NOT = { has_government = neutrality }
				}
				add_popularity = { ideology = neutrality popularity = -0.01 }
			}
			else_if = {
				limit = {
					check_variable = { steal_party_index > 19 }
					NOT = { has_government = nationalist }
				}
				add_popularity = { ideology = nationalist popularity = -0.01 }
			}
			set_temp_variable = { party_index = steal_party_index }
			set_temp_variable = { party_popularity_increase = -0.01 }
			add_relative_party_popularity = yes
			set_party_index_to_ruling_party = yes
			set_temp_variable = { party_popularity_increase = 0.01 }
			add_relative_party_popularity = yes
		}
		meta_effect = {
			text = {
				custom_effect_tooltip = attract_voters_pos.[INDEX]
			}
			INDEX = "[?steal_party_index|.0]"
		}
	}
}

set_party_index_to_ruling_party = {
	if = {
		limit = {
			is_in_array = { array = ruling_party value = 0 }
			NOT = { has_country_leader_with_trait = western_technocrat }
		}
		set_temp_variable = { party_index = 0 }
	}
	else_if = {
		limit = {
			is_in_array = { array = ruling_party value = 0 }
			has_country_leader_with_trait = western_technocrat
		}
		for_each_loop = {
			array = gov_coalition_array
			if = {
				limit = {
					check_variable = { party_pop_array^v > biggest_party_in_coalition_popularity }
				}
				set_temp_variable = { biggest_party_in_coalition_popularity = party_pop_array^v }
				set_temp_variable = { biggest_party_in_coalition = v }
			}
		}
		set_temp_variable = { party_index = biggest_party_in_coalition }
	}
	else = {
		set_temp_variable = { party_index = ruling_party^0 }
	}
}

# Add Relative Party Popularity:
#
# This scripted effect adds (or substracts) relative party popularity. How to use:
#
# 2022/04/30 - Added a additional parameter Z to allow you to increase the outlook within the script
#
# set_temp_variable = { party_index = X } #Index of party to be changed 0-23
# set_temp_variable = { party_popularity_increase = Y } #How much party popularity is changed, must be in decimals so 2% is 0.02
# set_temp_variable = { temp_outlook_increase = Z } #OPTIONAL PARAMETER -- Must be in decimals so 2% is 0.02
# add_relative_party_popularity = yes
#
#NOTE! The temp variable names must be the same as above!
add_relative_party_popularity = {
	custom_effect_tooltip = TT_ADD_RELATIVE_PARTY_POPULARITY
	if = {
		limit = { check_variable = { party_index < 4 } }

		# If you have the optional parameter. First ADD the new outlook percentage.
		if = { limit = { has_variable = temp_outlook_increase }
			add_popularity = { ideology = democratic popularity = temp_outlook_increase }
			add_to_variable = { party_pop_array^party_index = party_popularity_increase }
		}
		else = {
			#Set the old remainder Outlook based on non-boosted parties
			set_temp_variable = { old_outlook_remainder = party_popularity@democratic }
			subtract_from_temp_variable = { old_outlook_remainder = party_pop_array^party_index }
			#Add X% to party popularity, this is the popularity we want the party to retain!
			add_to_variable = { party_pop_array^party_index = party_popularity_increase }
			if = {
				limit = { check_variable = { old_outlook_remainder > 0 } }
				#Set the new remainder Outlook
				set_temp_variable = { outlook_remainder = party_popularity@democratic }
				subtract_from_temp_variable = { outlook_remainder = party_pop_array^party_index }
				#Scale all non-boosted parties
				divide_temp_variable = { outlook_remainder = old_outlook_remainder }
				for_loop_effect = {
					start = 0
					end = 4
					value = i
					if = {
						limit = { NOT = { check_variable = { i = party_index } } }
						multiply_variable = { party_pop_array^i = outlook_remainder }
					}
				}
			}
			else = {
				add_popularity = { ideology = democratic popularity = party_popularity_increase }
			}
		}

hold_elections = {
	# Technocrat Trait
	if = { limit = { has_country_leader_with_trait = western_technocrat }
		set_variable = { party_pop_array^0 = 0 } #technocrats don't participate in elections
	}
	add_up_party_totals = yes
	set_party_positions = yes
	if = { limit = { NOT = { has_country_flag = partyall_banned } }
		for_loop_effect = {
			end = party_pop_array^num
			value = v
			if = {
				limit = {
					# normal popularity entry is over threshold
					check_variable = { party_pop_array^v > election_threshold }
				}
				set_variable = { party_pop_elect_array^v = party_pop_array^v }
			}
			else = {
				set_variable = { party_pop_elect_array^v = 0 }
			}
		}
	}
	else = {
		# Should set the main election value and ignore the election threshold
		for_each_loop = {
			array = ruling_party
			set_variable = { party_pop_elect_array^i = party_pop_array^i }
		}
	}
	calculate_banned_parties = yes
	update_majority_threshold = yes
	find_two_highest_parties = yes
	update_government_coalition_strength = yes
	update_dirty_subideology = yes
}

add_up_party_totals = {

	set_variable = { pop_total_fascism = 0 }
	set_variable = { pop_total_neutrality = 0 }
	set_variable = { pop_total_nationalist = 0 }

	#Outlook popularity based on party popularities (removed the unnecessary loop)
	add_to_variable = { pop_total_democratic = party_pop_array^0 }

	add_to_variable = { pop_total_democratic = party_pop_array^2 }
	add_to_variable = { pop_total_democratic = party_pop_array^3 }

	add_to_variable = { pop_total_communism = party_pop_array^4 }

	add_to_variable = { pop_total_fascism = party_pop_array^10 }

	add_to_variable = { pop_total_neutrality = party_pop_array^12 }

	add_to_variable = { pop_total_neutrality = party_pop_array^14 }
	add_to_variable = { pop_total_neutrality = party_pop_array^15 }

	add_to_variable = { pop_total_neutrality = party_pop_array^18 }
	add_to_variable = { pop_total_neutrality = party_pop_array^19 }

	add_to_variable = { pop_total_nationalist = party_pop_array^20 }
	add_to_variable = { pop_total_nationalist = party_pop_array^21 }
	add_to_variable = { pop_total_nationalist = party_pop_array^22 }
	add_to_variable = { pop_total_nationalist = party_pop_array^23 }

	#Scale party popularity totals to match actual Outlook popularity
	for_loop_effect = {
		end = party_pop_array^num
		value = current_id
		multiply_variable = { party_pop_array^current_id = 1000 }
		if = {
			limit = { check_variable = { current_id < 4 } }
			if = {
				limit = { check_variable = { pop_total_democratic > 0 } }
				divide_variable = { party_pop_array^current_id = pop_total_democratic }
				multiply_variable = { party_pop_array^current_id = party_popularity@democratic }
			}
			else = {
				set_variable = { party_pop_array^current_id = 0 }
			}
		}
		else_if = {
			limit = { check_variable = { current_id < 10 } }
			if = {
				limit = { check_variable = { pop_total_communism > 0 } }
				divide_variable = { party_pop_array^current_id = pop_total_communism }
				multiply_variable = { party_pop_array^current_id = party_popularity@communism }
			}
			else = {
				set_variable = { party_pop_array^current_id = 0 }
			}
		}
		else_if = {
			limit = { check_variable = { current_id < 12 } }
			if = {
				limit = { check_variable = { pop_total_fascism > 0 } }
				divide_variable = { party_pop_array^current_id = pop_total_fascism }
				multiply_variable = { party_pop_array^current_id = party_popularity@fascism }
			}
			else = {
				set_variable = { party_pop_array^current_id = 0 }
			}
		}
		else_if = {
			limit = { check_variable = { current_id < 20 } }
			if = {
				limit = { check_variable = { pop_total_neutrality > 0 } }
				divide_variable = { party_pop_array^current_id = pop_total_neutrality }
				multiply_variable = { party_pop_array^current_id = party_popularity@neutrality }
			}
			else = {
				set_variable = { party_pop_array^current_id = 0 }
			}
		}
		else = {
			if = {
				limit = { check_variable = { pop_total_nationalist > 0 } }
				divide_variable = { party_pop_array^current_id = pop_total_nationalist }
				multiply_variable = { party_pop_array^current_id = party_popularity@nationalist }
			}
			else = {
				set_variable = { party_pop_array^current_id = 0 }
			}
		}
		multiply_variable = { party_pop_array^current_id = 0.001 }
		clamp_variable = { var = party_pop_array^current_id min = 0 max = 1 }
	}

	#Make sure there are no Outlooks with popularity without any party popularities
	#Make 1% the limit, as smaller than that isn't visible on the chart
	if = {
		limit = {
			democratic > 0.01
			check_variable = { pop_total_democratic < 0.01 }
		}
		random_list = {
			10 = { add_to_variable = { party_pop_array^0 = 0.01 } }

			10 = { add_to_variable = { party_pop_array^2 = 0.01 } }

		}
	}
	if = {

			communism > 0.01
			check_variable = { pop_total_communism < 0.01 }
		}
		random_list = {
			10 = { add_to_variable = { party_pop_array^4 = 0.01 } }
			10 = { add_to_variable = { party_pop_array^5 = 0.01 } }
			10 = { add_to_variable = { party_pop_array^6 = 0.01 } }

			10 = { modifier = { factor = 0 NOT = { has_idea = shia } } add_to_variable = { party_pop_array^9 = 0.01 } }
		}
	}
	if = {
		limit = {
			fascism > 0.01
			check_variable = { pop_total_fascism < 0.01 }
		}
		if = {
			limit = { Is_Muslim_Country = yes }
			random_list = {
				10 = { add_to_variable = { party_pop_array^10 = 0.01 } }
				10 = { add_to_variable = { party_pop_array^11 = 0.01 } }
			}
		}
	}
	if = {
		limit = {
			neutrality > 0.01
			check_variable = { pop_total_neutrality < 0.01 }
		}
		random_list = {
			10 = { add_to_variable = { party_pop_array^12 = 0.01 } }

set_party_positions = {
	resize_temp_array = {
		array = party_pop_array_temp
		size = 24
	}
	for_loop_effect = {
		end = pos_array^num
		find_highest_in_array = {
			array = party_pop_array
			value = max
			index = max_index
		}
		set_variable = { pos_array^v = max_index }
		set_variable = { party_pop_array^max_index = -1 }
		set_temp_variable = { party_pop_array_temp^max_index = max }
	}
	clear_array = party_pop_array

		array = party_pop_array_temp
		add_to_array = { party_pop_array = v }
	}
	clear_temp_array = party_pop_array_temp
}

# Function: gov_pop_calculation
# Purpose: The gov pop calculation is to calculate the penalty from your government popularity swinging
# from -100% to 100% of their current popularity
# How-to Adjust: Adjust the penalty by changing line 1904 from 0.002 to 0.005
gov_pop_calculation = {
	if = { limit = { NOT = { has_dynamic_modifier = { modifier = government_popularity_modifier } } }
		add_dynamic_modifier = { modifier = government_popularity_modifier }
	}
	set_temp_variable = { add_t = government_coalition_differential }
	multiply_temp_variable = { add_t = 100 }
	clamp_variable = { var = add_t max = 100 min = -100 }
	multiply_temp_variable = { add_t = 0.002 }

	set_variable = { gov_pop_stability_factor = add_t }
	set_variable = { gov_pop_political_power_factor = add_t }
	set_variable = { gov_pop_drift_defence_factor = add_t }
}

set_coalition_drift = {
	#reset coalition counts
	set_variable = { coalition_fascism = 0 }
	set_variable = { coalition_nationalist = 0 }
	set_variable = { coalition_communism = 0 }
	set_variable = { coalition_neutrality = 0 }
	set_variable = { coalition_democratic = 0 }
	clear_coalition_count_ideas = yes

	#Each coalition partner popularity gets added to their own Outlook
	for_loop_effect = {
		end = gov_coalition_array^num
		value = v
		set_temp_variable = { sum = party_pop_elect_array^gov_coalition_array^v } #pop of current element
		set_temp_variable = { current_id = gov_coalition_array^v } #ID of current element
		if = {
			limit = {
				check_variable = { current_id < 4 }
			}
			add_to_variable = { coalition_democratic = sum }
		}
		else_if = {
			limit = {
				check_variable = { current_id > 3 }
				check_variable = { current_id < 10 }
			}
			add_to_variable = { coalition_communism = sum }
		}
		else_if = {
			limit = {
				OR = {
					check_variable = { current_id = 10 }
					check_variable = { current_id = 11 }
				}
			}
			add_to_variable = { coalition_fascism = sum }
		}

# Transfers coalition leadership to the largest coalition member if they outrank the ruling party.
# Rebuilds the government via change_ruling_party_effect to ensure leader/ideology swap.
# Called after coalition re-election (MD4_election.2).
update_coalition_leader = {
	if = {
		limit = { check_variable = { gov_coalition_array^num > 0 } }

		# Build temp array of all coalition members (ruling party + partners)
		clear_temp_array = full_coalition_temp
		add_to_temp_array = { full_coalition_temp = ruling_party^0 }
		for_each_loop = {
			array = gov_coalition_array
			value = v
			add_to_temp_array = { full_coalition_temp = v }
		}

		# Find the strongest member by electoral popularity
		set_temp_variable = { best_coalition_pop = 0 }
		set_temp_variable = { best_coalition_index = ruling_party^0 }
		for_each_loop = {
			array = full_coalition_temp
			value = v
			if = {
				limit = { check_variable = { party_pop_elect_array^v > best_coalition_pop } }
				set_temp_variable = { best_coalition_pop = party_pop_elect_array^v }
				set_temp_variable = { best_coalition_index = v }
			}
		}

		# Only swap if a different party is now strongest
		if = {
			limit = {
				NOT = { is_in_array = { ruling_party = best_coalition_index } }
			}

			# Set up change_ruling_party_effect parameters
			set_temp_variable = { rul_party_temp = best_coalition_index }
			set_temp_variable = { disable_elections = 0 }

			# Assign remaining coalition members to col slots
			set_temp_variable = { coalition_slot = 0 }
			for_each_loop = {
				array = full_coalition_temp
				value = v
				if = {
					limit = {
						NOT = { check_variable = { v = best_coalition_index } }
						check_variable = { party_pop_elect_array^v > election_threshold }
					}
					if = { limit = { check_variable = { coalition_slot = 0 } }
						set_temp_variable = { col_one = v }
					}
					else_if = { limit = { check_variable = { coalition_slot = 1 } }
						set_temp_variable = { col_two = v }
					}
					else_if = { limit = { check_variable = { coalition_slot = 2 } }
						set_temp_variable = { col_three = v }
					}
					else_if = { limit = { check_variable = { coalition_slot = 3 } }
						set_temp_variable = { col_four = v }
					}
					else_if = { limit = { check_variable = { coalition_slot = 4 } }
						set_temp_variable = { col_five = v }
					}
					add_to_temp_variable = { coalition_slot = 1 }
				}
			}

			change_ruling_party_effect = yes
		}
	}
}

	if = {
		limit = { NOT = { check_variable = { global.pol_dist_initialized = 1 } } }
		init_pol_distance_matrix = yes
	}

}

# Function: Change Party Command
# Parameters:
# set_temp_variable = { rul_party_temp = 0-23 }
# set_temp_variable = { col_one = 0 - 23 }  -- Optional
# set_temp_variable = { col_two = 0 - 23 }  -- Optional
# set_temp_variable = { col_three = 0 - 23 }  -- Optional
# set_temp_variable = { col_four = 0 - 23 }  -- Optional
# set_temp_variable = { col_five = 0 - 23 }  -- Optional
# set_temp_variable = { change_leader_temp = 0-1 } -- Optional
# set_temp_variable = { disable_elections = 0-1 } -- Optional
change_ruling_party_effect = {
	# Do Not Retire Section
	if = { limit = { check_variable = { change_leader_temp = 1 } }
		set_country_flag = do_not_retire
	}

	# Clears the Parameters
	clear_array = ruling_party
	clear_array = gov_coalition_array

	# Rul Party Addition
	custom_effect_tooltip = ruling_party_changed_tt
	add_to_array = { ruling_party = rul_party_temp }

	# Coalition Array Parameters
	if = { limit = { has_variable = col_one }
		set_temp_variable = { add_col_one = col_one }
		custom_effect_tooltip = add_coalition_members_effect_TT
		add_to_array = { gov_coalition_array = add_col_one }
	}
	if = { limit = { has_variable = col_two }
		set_temp_variable = { add_col_one = col_two }
		custom_effect_tooltip = add_coalition_members_effect_TT
		add_to_array = { gov_coalition_array = add_col_one }
	}
	if = { limit = { has_variable = col_three }
		set_temp_variable = { add_col_one = col_three }
		custom_effect_tooltip = add_coalition_members_effect_TT
		add_to_array = { gov_coalition_array = add_col_one }
	}
	if = { limit = { has_variable = col_four }
		set_temp_variable = { add_col_one = col_four }
		custom_effect_tooltip = add_coalition_members_effect_TT
		add_to_array = { gov_coalition_array = add_col_one }
	}
	if = { limit = { has_variable = col_five }
		set_temp_variable = { add_col_one = col_five }
		custom_effect_tooltip = add_coalition_members_effect_TT
		add_to_array = { gov_coalition_array = add_col_one }
	}

	# Disable Elections of 0 means that they have elections
	if = { limit = { check_variable = { disable_elections = 0 } }
		custom_effect_tooltip = ruling_party_changed_elections_tt
		if = {
			limit = {
				check_variable = { rul_party_temp > -1 }
				check_variable = { rul_party_temp < 4 }
			}
			hidden_effect = {
				set_politics = {
					ruling_party = democratic
					elections_allowed = yes
				}
			}
		}
		else_if = {
			limit = {
				check_variable = { rul_party_temp > 3 }
				check_variable = { rul_party_temp < 10 }
			}
			hidden_effect = {
				set_politics = {
					ruling_party = communism
					elections_allowed = yes
				}
			}
		}
		else_if = {
			limit = {
				check_variable = { rul_party_temp > 9 }
				check_variable = { rul_party_temp < 12 }
			}
			hidden_effect = {
				set_politics = {
					ruling_party = fascism
					elections_allowed = yes
				}
			}
		}
		else_if = {
			limit = {
				check_variable = { rul_party_temp > 11 }
				check_variable = { rul_party_temp < 20 }
			}
			hidden_effect = {
				set_politics = {
					ruling_party = neutrality
					elections_allowed = yes
				}
			}
		}
		else_if = {
			limit = {
				check_variable = { rul_party_temp > 19 }
				check_variable = { rul_party_temp < 24 }
			}
			hidden_effect = {
				set_politics = {
					ruling_party = nationalist
					elections_allowed = yes
				}
			}
		}
	}
	else = {
		custom_effect_tooltip = ruling_party_changed_no_elections_tt
		if = {
			limit = {
				check_variable = { rul_party_temp > -1 }
				check_variable = { rul_party_temp < 4 }
			}
			hidden_effect = {
				set_politics = {
					ruling_party = democratic
					elections_allowed = no
				}
			}
		}
		else_if = {
			limit = {
				check_variable = { rul_party_temp > 3 }
				check_variable = { rul_party_temp < 11 }
			}
			hidden_effect = {
				set_politics = {
					ruling_party = communism
					elections_allowed = no
				}
			}
		}
		else_if = {
			limit = {
				check_variable = { rul_party_temp > 10 }
				check_variable = { rul_party_temp < 12 }
			}
			hidden_effect = {
				set_politics = {
					ruling_party = fascism
					elections_allowed = no
				}
			}
		}
		else_if = {
			limit = {
				check_variable = { rul_party_temp > 11 }
				check_variable = { rul_party_temp < 20 }
			}
			hidden_effect = {
				set_politics = {
					ruling_party = neutrality
					elections_allowed = no
				}
			}
		}
		else_if = {
			limit = {
				check_variable = { rul_party_temp > 19 }
				check_variable = { rul_party_temp < 24 }
			}
			hidden_effect = {
				set_politics = {
					ruling_party = nationalist
					elections_allowed = no
				}
			}
		}
	}

	hidden_effect = {
		update_government_coalition_strength = yes
		update_party_name = yes

		IP_idea_configuration = yes

	}

		end = 24

			}
			set_temp_variable = { ban_flag_index = ban_i }
			add_to_temp_variable = { ban_flag_index = 1 }

				text = {
					if = {
						limit = { NOT = { has_country_flag = party[v]_banned } }

				}
				v = "[?ban_flag_index|.0]"
			}
		}
	}
	if = { limit = { NOT = { has_country_flag = free_ban_parties } }
		multiply_temp_variable = { pp_cost = -1 }

# Effect: enable_elections_with_no_leader_change
# Purpose: Enable elections with no leader change
enable_elections_with_no_leader_change = {
	set_country_flag = no_leader_change
	# Set national term limit if there is such a system for your nation
	if = { limit = { check_variable = { term_limit > 0 } }
		set_variable = { current_term = 1 }
	}

	if = { limit = { has_government = democratic }
		set_politics = {
			ruling_party = democratic
			elections_allowed = yes
		}
	}
	else_if = { limit = { has_government = communism }
		set_politics = {
			ruling_party = communism
			elections_allowed = yes
		}
	}
	else_if = { limit = { has_government = fascism }
		set_politics = {
			ruling_party = fascism
			elections_allowed = yes
		}
	}
	else_if = { limit = { has_government = neutrality }
		set_politics = {
			ruling_party = neutrality
			elections_allowed = yes
		}
	}
	else_if = { limit = { has_government = nationalist }
		set_politics = {
			ruling_party = nationalist
			elections_allowed = yes
		}
	}
	clr_country_flag = no_leader_change
}
