every_possible_country = {
	limit = {
		OR = {
			nuclear_facility > 0
			naval_facility > 0
			air_facility > 0
			land_facility > 0
		}
	}

	if = {
		limit = {
			nuclear_facility > 0
		}
		generate_scientist_character = {
			gender = male
			skills = {
				specialization_nuclear = 1
			}
		}
	}

	if = {
		limit = {
			naval_facility > 0
		}
		generate_scientist_character = {
			gender = male
			skills = {
				specialization_naval = 1
			}
		}
	}

	if = {
		limit = {
			air_facility > 0
		}
		generate_scientist_character = {
			gender = male
			skills = {
				specialization_air = 1
			}
		}
	}

	if = {
		limit = {
			land_facility > 0
		}
		generate_scientist_character = {
			gender = male
			skills = {
				specialization_land = 1
			}
		}
	}
}
