EU_is_leader_of_foreign_policy = {
	has_idea = EU_member
	OR = {
		has_idea = EU_foreign_minister
		AND = {
			check_expr = { value = global.euforeignaffairs equals = { value = EUU.id } }
			has_idea = EU_commission_president
		}
		AND = {
			check_expr = { value = global.euforeignaffairs equals = { value = EUU.id } }
			check_expr = { value = global.eucomission equals = { value = EUU.id } }
			has_idea = EU_parliament_president
		}
	}
}

EU_is_leader_of_trade_policy = {
	has_idea = EU_member
	OR = {
		has_idea = EU_commission_president
		AND = {
			check_expr = { value = global.eucomission equals = { value = EUU.id } }
			has_idea = EU_council_president
		}
		AND = {
			check_expr = { value = global.eucomission equals = { value = EUU.id } }
			check_expr = { value = global.eucouncil equals = { value = EUU.id } }
			has_idea = EU_parliament_president
		}
	}
}

EU_influence_on_leader_of_foreign_policy_25_percent = {
	custom_trigger_tooltip = {
		tooltip = EU_influence_on_leader_of_foreign_policy_25_percent_TT
		OR = {
			# A foreign-policy office holder exists: negotiate exit with them.
			AND = {
				any_of_scopes = {
					array = global.EU_potential
					EU_is_leader_of_foreign_policy = yes
				}
				OR = {
					# You hold EU foreign policy: negotiate your own exit
					EU_is_leader_of_foreign_policy = yes
					any_of_scopes = {
						array = global.EU_member
						EU_is_leader_of_foreign_policy = yes
						influence_higher_25 = yes
					}
				}
			}
			# No EU foreign-policy office is filled (foreign minister, commission president and
			# parliament president are all vacant — they are 5-year timed ideas that can lapse).
			# Fall back to the broad-influence bar so exit is never hard-locked on an office that
			# does not currently exist.
			AND = {
				NOT = {
					any_of_scopes = {
						array = global.EU_potential
						EU_is_leader_of_foreign_policy = yes
					}
				}
				EU_influence_on_all_other_members_5_percent = yes
			}
		}
	}
}

stop_pro_european_campaign = {
	OR = {
		date < 2000.01.15
		AND = {
			check_expr = { value = global.var_europeanism greater_than_or_equals = { value = 0.85 } }
			check_expr = { value = THIS.europeanism less_than = { value = 0.25 } }
		}
		AND = {
			check_expr = { value = THIS.europeanism less_than = { value = 0.25 } }
			NOT = {
				any_of_scopes = {
					array = global.EU_member
					check_expr = { value = THIS.europeanism greater_than = { value = 0.50 } }
				}
			}
		}
	}
}
