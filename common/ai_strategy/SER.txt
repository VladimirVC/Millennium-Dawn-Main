#I hate Kosovo
SER_hate_KOS = {
	allowed = { original_tag = SER }
	enable = {
		country_exists = KOS
		NOT = { KOS = { is_subject_of = SER } }
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = antagonize id = "KOS" value = 150 }
	ai_strategy = { type = conquer id = "KOS" value = 150 }
}
#Russian Friend
SER_befriend_SOV = {
	allowed = { original_tag = SER }
	enable = {
		SOV = { NOT = { has_government = democratic } }
		NOT = { has_government = democratic }
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = befriend id = "SOV" value = 150 }
}
#Help Srpska

#Democratic Serbia leans west
SER_liberal_eu_lean = {
	allowed = { original_tag = SER }
	enable = {
		OR = {
			has_global_flag = SER_SERBIAN_LIBERALS_FOCUS_PATH
			has_country_flag = SER_path_liberal
		}
		has_government = democratic
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = befriend id = "GER" value = 100 }
	ai_strategy = { type = befriend id = "FRA" value = 100 }
	ai_strategy = { type = befriend id = "ITA" value = 75 }
	ai_strategy = { type = alliance id = "SOV" value = -100 }
}
#Keep Montenegro in the fold
SER_protect_montenegro = {
	allowed = { original_tag = SER }
	enable = {
		OR = {
			has_global_flag = SER_MILOCEVIC_NATI_FOCUS_PATH
			has_country_flag = SER_path_nationalist
		}
		country_exists = MNT
		check_variable = { SER_Montenegro_separatism < 50 }
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = protect id = "MNT" value = 150 }
	ai_strategy = { type = befriend id = "MNT" value = 100 }
}
