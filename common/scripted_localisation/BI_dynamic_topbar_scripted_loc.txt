# Scripted localisation for the tokenized topbar counter widget.
# Value text resolves from the slot's token key. See common/scripted_guis/BI_dynamic_topbar.txt.

defined_text = {
	name = GetSelectedTopbarInfoText
	text = {
		localization_key = [?topbar.GetTokenKey]_text
	}
}

defined_text = {
	name = GetCasualtiesAmount
	text = {
		trigger = {
			casualties > 1000000
			set_temp_variable = { casualties_k_k = casualties }
			divide_temp_variable = { casualties_k_k = 1000000 }
		}
		localization_key = TOPBAR_casualties_m
	}
	text = {
		trigger = {
			casualties > 100000
			set_temp_variable = { casualties_k = casualties }
			multiply_temp_variable = { casualties_k = 0.001 }
		}
		localization_key = TOPBAR_casualties_k
	}
	text = {
		localization_key = TOPBAR_casualties
	}
}

defined_text = {
	name = GetTopbarAllianceStrengthRatio
	text = {
		trigger = {
			has_war = yes
			check_variable = { alliance_strength_ratio > 1 }
		}
		localization_key = "[?alliance_strength_ratio|G2]"
	}
	text = {
		trigger = {
			has_war = yes
			check_variable = { alliance_strength_ratio < 1 }
			check_variable = { alliance_strength_ratio > 0 }
		}
		localization_key = "[?alliance_strength_ratio|Y2]"
	}
	text = {
		trigger = {
			has_war = yes
			check_variable = { alliance_strength_ratio < 0 }
		}
		localization_key = "[?alliance_strength_ratio|R2]"
	}
	text = {
		trigger = {
			has_war = no
		}
		localization_key = "-"
	}
}
