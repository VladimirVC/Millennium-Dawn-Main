### Mažeikiai Refinery Privatization ###

country_event = {
	id = lithuania.1
	title = lithuania.1.t
	desc = lithuania.1.d
	picture = GFX_oil_refinery
	fire_only_once = yes
	is_triggered_only = yes

	# Sell to PKN Orlen (Poland)
	option = {
		name = lithuania.1.a
		log = "[GetDateText]: [This.GetName]: lithuania.1.a executed"
		set_country_flag = LIT_sold_to_poland
		add_timed_idea = { idea = LIT_refinery_sale_revenue days = 365 }
		add_opinion_modifier = { target = POL modifier = large_increase }
		hidden_effect = {
			POL = {
				add_opinion_modifier = { target = LIT modifier = large_increase }
				country_event = { id = lithuania.2 days = 5 }
			}
			news_event = { id = lithuania_news.1 days = 2 }
		}
		reverse_add_opinion_modifier = { target = RUS modifier = medium_decrease }
		ai_chance = { base = 70 }
	}

	# Sell to Russian consortium
	option = {
		name = lithuania.1.b
		log = "[GetDateText]: [This.GetName]: lithuania.1.b executed"
		set_country_flag = LIT_sold_to_russia
		add_timed_idea = { idea = LIT_refinery_sale_revenue days = 365 }
		add_ideas = LIT_russian_energy_leverage
		add_opinion_modifier = { target = RUS modifier = medium_increase }
		reverse_add_opinion_modifier = { target = POL modifier = medium_decrease }
		hidden_effect = {
			country_event = { id = lithuania.3 days = 120 random_days = 90 }
			news_event = { id = lithuania_news.1 days = 2 }
		}
		ai_chance = { base = 15 }
	}

	# Keep it nationalized
	option = {
		name = lithuania.1.c
		log = "[GetDateText]: [This.GetName]: lithuania.1.c executed"
		add_political_power = -50
		add_stability = -0.03
		add_ideas = LIT_nationalized_refinery
		ai_chance = { base = 15 }
	}
}

### Ignalina Nuclear Power Plant ###

country_event = {
	id = lithuania.10
	title = lithuania.10.t
	desc = lithuania.10.d
	picture = GFX_nuclear_power_plant
	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		controls_state = 109
	}

	# Negotiate with Poland and Baltic States
	option = {
		name = lithuania.10.a
		log = "[GetDateText]: [This.GetName]: lithuania.10.a executed"
		add_political_power = -25
		hidden_effect = {
			POL = { country_event = { id = lithuania.11 days = 14 } }
		}
		ai_chance = { base = 70 }
	}

	# Keep it running
	option = {
		name = lithuania.10.b
		log = "[GetDateText]: [This.GetName]: lithuania.10.b executed"
		set_country_flag = LIT_ignalina_kept_open
		add_stability = -0.05
		reverse_add_opinion_modifier = { target = POL modifier = small_decrease }
		reverse_add_opinion_modifier = { target = LAT modifier = small_decrease }
		reverse_add_opinion_modifier = { target = EST modifier = small_decrease }
		hidden_effect = {
			country_event = { id = lithuania.16 days = 730 }
		}
		ai_chance = { base = 15 }
	}

	# Close without replacement
	option = {
		name = lithuania.10.c
		log = "[GetDateText]: [This.GetName]: lithuania.10.c executed"
		remove_ideas = nuclear_energy
		add_ideas = LIT_energy_deficit
		hidden_effect = {
			news_event = { id = lithuania_news.2 days = 3 }
		}
		ai_chance = { base = 15 }
	}
}

country_event = {
	id = lithuania.14
	title = lithuania.14.t
	desc = lithuania.14.d
	picture = GFX_nuclear_power_plant
	is_triggered_only = yes

	# People approve
	option = {
		name = lithuania.14.a
		log = "[GetDateText]: [This.GetName]: lithuania.14.a executed"
		add_stability = 0.05
		if = {
			limit = { has_idea = LIT_energy_deficit }
			remove_ideas = LIT_energy_deficit
		}
		if = {
			limit = { NOT = { has_idea = nuclear_energy } }
			add_ideas = nuclear_energy
		}
		add_ideas = LIT_visaginas_nuclear_plant
		hidden_effect = {
			POL = { country_event = { id = lithuania.15 days = 3 } }
			news_event = { id = lithuania_news.3 days = 5 }
		}
		ai_chance = { base = 40 }
	}

	# People reject
	option = {
		name = lithuania.14.b
		log = "[GetDateText]: [This.GetName]: lithuania.14.b executed"
		add_stability = -0.05
		add_political_power = -50
		add_ideas = LIT_energy_deficit
		ai_chance = { base = 60 }
	}
}

	id = lithuania_news.1
	title = lithuania_news.1.t
	desc = {
		text = lithuania_news.1.d_poland
		trigger = { LIT = { has_country_flag = LIT_sold_to_poland } }
	}
	desc = {
		text = lithuania_news.1.d_russia
		trigger = { LIT = { has_country_flag = LIT_sold_to_russia } }
	}
	picture = GFX_oil_refinery
	major = yes
	is_triggered_only = yes

	option = {
		name = lithuania_news.1.a
		trigger = { original_tag = LIT }
		hidden_effect = {
			clr_country_flag = LIT_sold_to_poland
			clr_country_flag = LIT_sold_to_russia
		}
	}

	option = {
		name = lithuania_news.1.b
		trigger = { original_tag = POL }
	}

	option = {
		name = lithuania_news.1.c
		trigger = { original_tag = RUS }
	}

	option = {
		name = lithuania_news.1.d
		trigger = {
			NOT = { original_tag = LIT }
			NOT = { original_tag = POL }
			NOT = { original_tag = RUS }
			capital_scope = { is_on_continent = europe }
		}
	}

	option = {
		name = lithuania_news.1.e
