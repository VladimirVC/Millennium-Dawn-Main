country_event = {
	id = MD_antarctica.2
	hidden = yes
	is_triggered_only = yes

	immediate = {
		clr_country_flag = antarctica_observer_mission_in_progress

		if = {
			limit = { check_variable = { antarctica_observer_target_station > 0 } }

			meta_effect = {
				text = {
					clr_global_flag = research_station_[ID]_observers_incoming
				}
				ID = "[?antarctica_observer_target_station]"
			}

			meta_effect = {
				text = {
					set_temp_variable = { station_illegal_activities = global.research_station_[ID]_illegal_activities_enabled }
					set_temp_variable = { station_open_loop_module = global.research_station_[ID]_module_slot_12 }
					set_temp_variable = { station_lab_slot_6 = global.research_station_[ID]_module_slot_6 }
					set_temp_variable = { station_lab_slot_7 = global.research_station_[ID]_module_slot_7 }
					set_temp_variable = { station_lab_slot_8 = global.research_station_[ID]_module_slot_8 }
					set_temp_variable = { station_lab_slot_9 = global.research_station_[ID]_module_slot_9 }
				}
				ID = "[?antarctica_observer_target_station]"
			}

			if = {
				limit = {
					OR = {
						check_variable = { station_illegal_activities > 0 }
						check_variable = { station_open_loop_module = 1 }
					}
				}
				add_to_variable = { antarctica_illegal_research_reports = 1 }

				# Violation precedence: nukes research > illegal research > open-loop utilities.
				if = {
					limit = {
						check_variable = { station_illegal_activities > 0 }
						OR = {
							check_variable = { station_lab_slot_6 = 7 }
							check_variable = { station_lab_slot_7 = 7 }
							check_variable = { station_lab_slot_8 = 7 }
							check_variable = { station_lab_slot_9 = 7 }
						}
					}
					if = {
						limit = { check_variable = { antarctica_observer_target_country > 0 } }
						var:antarctica_observer_target_country = {
							antarctica_observer_detected_nukes_research_effect = yes
						}
					}
				}
				else_if = {
					limit = { check_variable = { station_illegal_activities > 0 } }
					if = {
						limit = { check_variable = { antarctica_observer_target_country > 0 } }
						var:antarctica_observer_target_country = {
							antarctica_observer_detected_illegal_research_effect = yes
						}
					}
				}
				else_if = {
					limit = { check_variable = { station_open_loop_module = 1 } }
					if = {
						limit = { check_variable = { antarctica_observer_target_country > 0 } }
						var:antarctica_observer_target_country = {
							antarctica_observer_detected_open_loop_effect = yes
						}
					}
				}
			}
		}
	}
}
