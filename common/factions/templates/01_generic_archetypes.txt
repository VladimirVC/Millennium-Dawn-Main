# Generic faction templates for NCNS compatibility
# These templates are visible in the faction creation UI for countries
# without dedicated 99_*.txt templates, based on ideology alignment.

	name = GENERIC_WESTERN_ALLIANCE
	icon = GFX_faction_logo_generic_democratic
	manifest = faction_manifest_defense_of_democracy
	can_leader_join_other_factions = yes
	visible = {
		has_government = democratic
	}

	goals = {
		# Short term
		faction_goal_growing_the_faction
		faction_goal_unlock_research

		faction_goal_economic_integration
		faction_goal_regional_stability
		# Long term
		faction_goal_semiconductor_supremacy
		faction_goal_defeat_of_communism

	default_rules = {
		# Joining

		# Call To War
		call_to_war_rule_defensive_only
		# Dismissal
		dismissal_rule_offensive_war
	}
}
