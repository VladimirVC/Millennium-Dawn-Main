ZOM_check_and_execute_spawn = {
	if = {
		limit = { NOT = { has_global_flag = ZOM_apocalypse_spawned } }
		set_global_flag = ZOM_apocalypse_spawned
		ZOM_execute_zombie_spawn = yes
	}
}
