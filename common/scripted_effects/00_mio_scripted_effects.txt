# Effect: free_trait_picks_used
# Purpose: Displays the number of free_trait_picks for a MIO to ensure there is a display and an notification to the player to make it easier to know when and why it's free
free_trait_picks_used = {
	custom_effect_tooltip = number_of_mio_free_trait_picks
	add_to_variable = { free_trait_picks = -1 }

	if = { limit = { check_variable = { free_trait_picks = 0 } }
		clear_variable = free_trait_picks
	}
}

expenditure_for_mio_upgrade = {
	if = { limit = { check_variable = { free_trait_picks > 0 } }
		free_trait_picks_used = yes
	}
	else = {
		FROM = {
			small_expenditure = yes
		}
	}
}

# MIO Unlock Catalog — master token array seeded in setup_global_arrays (00_startup_effects.txt).

update_mio_catalog_dirty_variable = {
	if = { limit = { check_variable = { global.mio_catalog_dirty_update_var < 10000 } }
		add_to_variable = { global.mio_catalog_dirty_update_var = 1 }
	}
	else = {
		set_variable = { global.mio_catalog_dirty_update_var = 1 }
	}
}

mio_catalog_rebuild_visible_yes = {
	clear_array = mio_catalog_visible_array
	set_temp_variable = { mio_catalog_iter = 1 }
	while_loop_effect = {
		limit = { check_variable = { mio_catalog_iter < 24 } }
		set_temp_variable = { v = mio_catalog_iter }
		if = {
			limit = { mio_catalog_entry_available_yes = yes }
			add_to_array = { mio_catalog_visible_array = mio_catalog_iter }
		}
		add_to_temp_variable = { mio_catalog_iter = 1 }
	}
}

ITA_mio_accumulate_merge_sums = {
	set_variable = { ITA_mio_merge_counter = 0 }
	while_loop_effect = {
		limit = {
			check_variable = { ITA_mio_merge_counter < 100 }
			has_mio_number_of_completed_traits > var:ITA_mio_merge_counter
		}
		add_to_variable = { ITA_mio_merge_counter = 1 }
	}
	ROOT = {
		add_to_variable = { ITA_mio_merge_trait_sum = PREV.ITA_mio_merge_counter }
		add_to_variable = { ITA_mio_merge_picks_sum = PREV.ITA_mio_merge_counter }
	}
	if = {
		limit = { check_variable = { free_trait_picks > 0 } }
		ROOT = { add_to_variable = { ITA_mio_merge_picks_sum = PREV.free_trait_picks } }
	}
	clear_variable = ITA_mio_merge_counter
}

ITA_alenia_aermacchi_merger_carry_traits = {
	set_variable = { ITA_mio_merge_trait_sum = 0 }
	set_variable = { ITA_mio_merge_picks_sum = 0 }
	mio:ITA_alenia_aeronautica = { ITA_mio_accumulate_merge_sums = yes }
	mio:ITA_aermacchi_manufacturer = { ITA_mio_accumulate_merge_sums = yes }
	if = {
		limit = { check_variable = { ITA_mio_merge_trait_sum > 0 } }
		mio:ITA_alenia_aermacchi = { add_mio_size = ROOT.ITA_mio_merge_trait_sum }
	}
	if = {
		limit = { check_variable = { ITA_mio_merge_picks_sum > 0 } }
		mio:ITA_alenia_aermacchi = { add_to_variable = { free_trait_picks = ROOT.ITA_mio_merge_picks_sum } }
	}
	clear_variable = ITA_mio_merge_trait_sum
	clear_variable = ITA_mio_merge_picks_sum
}
