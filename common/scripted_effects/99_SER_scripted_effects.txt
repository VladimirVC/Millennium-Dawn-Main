modify_voj_separ_support = {
	add_to_variable = { voj_separ = modify_voj_separ }

	custom_effect_tooltip = SER_voj_separ_TT

	clamp_variable = {
		var = ROOT.voj_separ
		min = 0
		max = 100
	}
}

SER_apply_reconstruction_modifier = {
	if = {
		limit = { NOT = { has_dynamic_modifier = { modifier = SER_reconstruction_modifier } } }
		hidden_effect = { add_dynamic_modifier = { modifier = SER_reconstruction_modifier } }
		custom_effect_tooltip = { localization_key = adds_dynamic_modifier_tt MODIFIER = SER_reconstruction_modifier }
	}
	else = {
		custom_effect_tooltip = { localization_key = modifies_dynamic_modifier_tt MODIFIER = SER_reconstruction_modifier }
	}
}
