}

### BRI — target state selection wrapper ###
# 1=port  2=railway  3=pipeline  4=telecom  5=SEZ  6=mine  7=powerplant  8=prestige
# Picks one owned-controlled state matching the type, applies the matching state dynamic modifier
BRI_select_target_state = {
	if = {
		limit = { check_variable = { project_type = 1 } }
		random_owned_controlled_state = {
			limit = { is_coastal = yes }
			add_dynamic_modifier = { modifier = CHI_bri_state_infrastructure }
		}
		CHI = { add_to_variable = { CHI_bri_ports_built = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 2 } }
		random_owned_controlled_state = {
			limit = {
				any_neighbor_state = { CONTROLLER = { original_tag = CHI } }
			}
			add_dynamic_modifier = { modifier = CHI_bri_state_infrastructure }
		}
		CHI = { add_to_variable = { CHI_bri_railways_built = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 3 } }
		random_owned_controlled_state = {
			limit = {
				OR = {
					has_resources_amount = { resource = oil amount > 0 }
					has_resources_amount = { resource = steel amount > 0 }
					has_resources_amount = { resource = chromium amount > 0 }
				}
			}
			add_dynamic_modifier = { modifier = CHI_bri_state_infrastructure }
		}
		CHI = { add_to_variable = { CHI_bri_pipelines_built = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 4 } }
		CHI = { add_to_variable = { CHI_bri_telecom_built = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 5 } }
		random_owned_controlled_state = {
			limit = {
				OR = {
					is_coastal = yes
					infrastructure > 3
				}
			}
			add_dynamic_modifier = { modifier = CHI_bri_state_industrial }
		}
		CHI = { add_to_variable = { CHI_bri_industrial_zones = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 6 } }
		random_owned_controlled_state = {
			limit = {
				OR = {
					has_resources_amount = { resource = steel amount > 0 }
					has_resources_amount = { resource = chromium amount > 0 }
					has_resources_amount = { resource = aluminium amount > 0 }
					has_resources_amount = { resource = tungsten amount > 0 }
				}
			}
			add_dynamic_modifier = { modifier = CHI_bri_state_industrial }
		}
		CHI = { add_to_variable = { CHI_bri_mines_secured = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 7 } }
		random_owned_controlled_state = {
			limit = { industrial_complex > 0 }
			add_dynamic_modifier = { modifier = CHI_bri_state_industrial }
		}
		CHI = { add_to_variable = { CHI_bri_powerplants_built = 1 } }
	}
	else_if = {
		limit = { check_variable = { project_type = 8 } }
		random_owned_controlled_state = {
			limit = { is_capital = yes }
			add_dynamic_modifier = { modifier = CHI_bri_state_industrial }
		}
		CHI = { add_to_variable = { CHI_bri_prestige_projects = 1 } }
	}
}

# Maps CHI_demographic_aging to a dividend tier (0/1/2)
CHI_apply_demographic_tier = {
	if = {
		limit = { check_variable = { CHI_demographic_aging < 25 } }
		set_temp_variable = { CHI_demographic_target_tier = 0 }
	}
	else_if = {
		limit = { check_variable = { CHI_demographic_aging < 50 } }
		set_temp_variable = { CHI_demographic_target_tier = 1 }
	}
	else = {
		set_temp_variable = { CHI_demographic_target_tier = 2 }
	}

	if = {
		limit = { NOT = { check_variable = { CHI_demographic_tier_applied = CHI_demographic_target_tier } } }

		# Subtract the previously-applied tier's contribution.
		if = {
			limit = { check_variable = { CHI_demographic_tier_applied = 0 } }
			add_to_variable = { CHI_economy_consumer_goods_factor = 0.07 }
			subtract_from_variable = { CHI_economy_productivity_growth_modifier = 0.03 }
			subtract_from_variable = { CHI_economy_total_workforce_modifier = 0.05 }
			add_to_variable = { CHI_economy_monthly_population = 0.30 }
		}
		else_if = {
			limit = { check_variable = { CHI_demographic_tier_applied = 1 } }
			add_to_variable = { CHI_economy_consumer_goods_factor = 0.04 }
			subtract_from_variable = { CHI_economy_productivity_growth_modifier = 0.02 }
			subtract_from_variable = { CHI_economy_total_workforce_modifier = 0.03 }
			add_to_variable = { CHI_economy_monthly_population = 0.30 }
		}
		else_if = {
			limit = { check_variable = { CHI_demographic_tier_applied = 2 } }
			add_to_variable = { CHI_economy_consumer_goods_factor = 0.02 }
			subtract_from_variable = { CHI_economy_productivity_growth_modifier = 0.01 }
			subtract_from_variable = { CHI_economy_total_workforce_modifier = 0.01 }
			add_to_variable = { CHI_economy_monthly_population = 0.25 }
		}

		# Add the new tier's contribution. Sentinel 9 (crash) adds nothing.
		if = {
			limit = { check_variable = { CHI_demographic_target_tier = 0 } }
			custom_effect_tooltip = { localization_key = modifies_dynamic_modifier_tt MODIFIER = CHI_economy_modifier }
			subtract_from_variable = { CHI_economy_consumer_goods_factor = 0.07 tooltip = consumer_goods_factor_tt }
			add_to_variable = { CHI_economy_productivity_growth_modifier = 0.03 tooltip = productivity_growth_modifier_tt }
			add_to_variable = { CHI_economy_total_workforce_modifier = 0.05 tooltip = total_workforce_modifier_tt }
			subtract_from_variable = { CHI_economy_monthly_population = 0.30 tooltip = monthly_population_tt }
		}
		else_if = {
			limit = { check_variable = { CHI_demographic_target_tier = 1 } }
			custom_effect_tooltip = { localization_key = modifies_dynamic_modifier_tt MODIFIER = CHI_economy_modifier }
			subtract_from_variable = { CHI_economy_consumer_goods_factor = 0.04 tooltip = consumer_goods_factor_tt }
			add_to_variable = { CHI_economy_productivity_growth_modifier = 0.02 tooltip = productivity_growth_modifier_tt }
			add_to_variable = { CHI_economy_total_workforce_modifier = 0.03 tooltip = total_workforce_modifier_tt }
			subtract_from_variable = { CHI_economy_monthly_population = 0.30 tooltip = monthly_population_tt }
		}
		else_if = {
			limit = { check_variable = { CHI_demographic_target_tier = 2 } }
			custom_effect_tooltip = { localization_key = modifies_dynamic_modifier_tt MODIFIER = CHI_economy_modifier }
			subtract_from_variable = { CHI_economy_consumer_goods_factor = 0.02 tooltip = consumer_goods_factor_tt }
			add_to_variable = { CHI_economy_productivity_growth_modifier = 0.01 tooltip = productivity_growth_modifier_tt }
			add_to_variable = { CHI_economy_total_workforce_modifier = 0.01 tooltip = total_workforce_modifier_tt }
			subtract_from_variable = { CHI_economy_monthly_population = 0.25 tooltip = monthly_population_tt }
		}

		set_variable = { CHI_demographic_tier_applied = CHI_demographic_target_tier }
	}
}

	if = {
		limit = { check_variable = { CHI_military_self_sufficiency > 99 } }
		if = { limit = { has_idea = CHI_russian_arms_dependency } remove_ideas = CHI_russian_arms_dependency }
		if = { limit = { has_idea = CHI_russian_arms_dependency_2 } remove_ideas = CHI_russian_arms_dependency_2 }

		limit = { check_variable = { CHI_military_self_sufficiency > 74 } }

### Uyghur Threat: monthly tick ###
# Called from on_monthly when CHI controls state 592.
# Drifts uighur_threat based on state 592 resistance, then rolls for uprising events.
CHI_uighur_monthly_tick = {
	# Guard: skip if ETK controls state 592
	if = {
		limit = {
			NOT = { 592 = { is_controlled_by = ETK } }
		}

		### Drift step: resistance drives uighur_threat up or down ###
		if = {
			limit = { 592 = { resistance > 60 } }
			set_temp_variable = { modify_uighur_threat = 2 }
			CHI_modify_uighur_threat = yes
		}
		else_if = {
			limit = { 592 = { resistance > 39 } }
			set_temp_variable = { modify_uighur_threat = 1 }
			CHI_modify_uighur_threat = yes
		}
		else_if = {
			limit = { 592 = { NOT = { resistance > 14 } } }
			set_temp_variable = { modify_uighur_threat = -1 }
			CHI_modify_uighur_threat = yes
		}

		### Secondary push from sinicization ###
		if = {
			limit = {
				check_variable = {
					var = ETK_sinicization_var
					value = 60
					compare = greater_than
				}
			}
			set_temp_variable = { modify_uighur_threat = 1 }
			CHI_modify_uighur_threat = yes
		}
		else_if = {
			limit = {
				check_variable = {
					var = ETK_sinicization_var
					value = 20
					compare = less_than
				}
			}
			set_temp_variable = { modify_uighur_threat = -1 }
			CHI_modify_uighur_threat = yes
		}

		### Event roll: gated by cooldown flag ###
		if = {
			limit = {
				NOT = { has_country_flag = CHI_uighur_event_cd }
			}

			# Calm tier: uighur_threat 0..29
			if = {
				limit = {
					check_variable = {
						var = uighur_threat
						value = 30
						compare = less_than
					}
				}
				random = {
					chance = 8
					country_event = chi_uighur.5
					set_country_flag = { flag = CHI_uighur_event_cd value = 1 days = 60 }
				}
			}
			# Tense tier: uighur_threat 30..59
			else_if = {
				limit = {
					check_variable = {
						var = uighur_threat
						value = 60
						compare = less_than
					}
				}
				random_list = {
					60 = { }
					30 = {
						country_event = chi_uighur.1
						set_country_flag = { flag = CHI_uighur_event_cd value = 1 days = 60 }
					}
					10 = {
						country_event = chi_uighur.2
						set_country_flag = { flag = CHI_uighur_event_cd value = 1 days = 60 }
					}
				}
			}
			# Volatile tier: uighur_threat 60..84
			else_if = {
				limit = {
					check_variable = {
						var = uighur_threat
						value = 85
						compare = less_than
					}
				}
				random_list = {
					35 = { }
					25 = {
						country_event = chi_uighur.1
						set_country_flag = { flag = CHI_uighur_event_cd value = 1 days = 60 }
					}
					30 = {
						country_event = chi_uighur.2
						set_country_flag = { flag = CHI_uighur_event_cd value = 1 days = 60 }
					}
					10 = {
						country_event = chi_uighur.3
						set_country_flag = { flag = CHI_uighur_event_cd value = 1 days = 60 }
					}
				}
			}
			# Boiling tier: uighur_threat 85..100
			else = {
				random_list = {
					15 = { }
					15 = {
						country_event = chi_uighur.1
						set_country_flag = { flag = CHI_uighur_event_cd value = 1 days = 60 }
					}
					35 = {
						country_event = chi_uighur.2
						set_country_flag = { flag = CHI_uighur_event_cd value = 1 days = 60 }
					}
					35 = {
						country_event = chi_uighur.3
						set_country_flag = { flag = CHI_uighur_event_cd value = 1 days = 60 }
					}
				}
			}
		}
	}
}
