is_frontline_state = {
	is_controlled_by = ROOT
	any_neighbor_state = { controller = { has_war_with = ROOT } }
}

is_second_line_state = {
	is_controlled_by = ROOT
	NOT = { is_frontline_state = yes }
	any_neighbor_state = { is_frontline_state = yes }
}

is_third_line_state = {
	is_controlled_by = ROOT
	NOT = {
		is_frontline_state = yes
		is_second_line_state = yes
	}
	any_neighbor_state = { is_second_line_state = yes }
}
