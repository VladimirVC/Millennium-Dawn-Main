give_AI_templates = {
	if = {
		limit = {
			OR = {
				original_tag = USA
				original_tag = SOV
				original_tag = CHI
			}
		}
		if = {
			limit = { NOT = { has_template = "AI Light Infantry Brigade" } }
			generic_AI_templates = yes
		}
		if = {
			limit = { NOT = { has_template = "AI Armored Division" } }
			Major_Power_AI_templates = yes
		}
	}
	else = {
		if = {
			limit = { NOT = { has_template = "AI Militia Brigade" } }
			Militia_AI_templates = yes
		}
		if = {
			limit = { NOT = { has_template = "AI Light Infantry Brigade" } }
			generic_AI_templates = yes
		}
	}

	if = {
		limit = {
			has_game_rule = {
				rule = rule_god_of_war
				option = yes
			}
			NOT = { has_template = "Mechanized Infantry Division" }
		}
		god_of_war_templates = yes
	}
}
