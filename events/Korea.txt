# Cooldown-gated: skips opening a new session if NKO_spa_recent_session is
# still active (i.e., another .10 fired within the last 100 days). Always
# reschedules so the cycle doesn't die.

	id = NKO_spa.10
	title = NKO_spa.10.t
	desc = NKO_spa.10.d

	is_triggered_only = yes
	hidden = yes

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					original_tag = NKO
					has_idea = NKO_spa_umbrella
					NOT = { has_civil_war = yes }
					NOT = { has_country_flag = NKO_spa_recent_session }
				}
				if = {
					limit = { check_variable = { NKO_spa_picks_made < 2 } }
					NKO_spa_force_pick_fallback = yes
					country_event = { id = NKO_spa.5 hours = 4 }

				country_event = { id = NKO_spa.1 hours = 6 }
				set_country_flag = { flag = NKO_spa_recent_session days = 100 value = 1 }
			}
			# Always reschedule so the cycle survives skipped iterations.
			country_event = { id = NKO_spa.10 days = 120 }
		}
	}

	option = {
		name = NKO_spa.10.a
		log = "[GetDateText]: [Root.GetName]: NKO_spa.10.a executed"

	hidden = yes
	is_triggered_only = yes

			limit = {
				has_country_flag = NKO_naval_buildup_unlocked
				NOT = { has_country_flag = NKO_naval_buildup_capstone_done }
				NOT = { has_country_flag = NKO_naval_buildup_recent_incident }
				NOT = { has_civil_war = yes }
			}
			set_country_flag = {
				flag = NKO_naval_buildup_recent_incident
				days = 30
				value = 1
			}
			random_list = {
				20 = { country_event = { id = nko_naval_buildup.51 days = 0 } }
				20 = { country_event = { id = nko_naval_buildup.52 days = 0 } }
				20 = { country_event = { id = nko_naval_buildup.53 days = 0 } }
				15 = { country_event = { id = nko_naval_buildup.54 days = 0 } }

				10 = { country_event = { id = nko_naval_buildup.56 days = 0 } }
			}
		}
		if = {
			limit = {
				has_country_flag = NKO_naval_buildup_unlocked
				NOT = { has_country_flag = NKO_naval_buildup_capstone_done }
			}
			country_event = { id = nko_naval_buildup.50 days = 60 }

	}

# Hooked from on_research_complete in NKO_naval_on_actions.txt.
# Bounty (in immediate, runs before the player sees the event):
#   - 5 hulls of each into the stockpile
#   - hard-coded Choe Hyon-class and Sinpo-class variants
#   - 2-factory production line for each at 50% on the first ship
# Player click rewards (in option): navy XP, manpower, war support.

	id = nko_naval_buildup.74

	# Re-check the trigger here so the event won't fire if the on_action
	# raced ahead of one of the techs finishing.
	trigger = {
		original_tag = NKO
		NOT = { has_country_flag = NKO_naval_modern_fleet_unlocked }
		has_tech = destroyer_hull_3
		has_tech = attack_submarine_hull_3
	}

	immediate = {
		set_country_flag = NKO_naval_modern_fleet_unlocked

		# ---------------------------------------------------------------
		# Hard-coded variants. Mid-grade modern loadout — competent at

		# leaves room for the player to upgrade radar / VLS / sonar /
		# torpedoes to the highest tier later via the Naval Designer.
		# ---------------------------------------------------------------

		# Choe Hyon-class destroyer — based on destroyer_hull_3.
		# Slots downgraded vs. the hull default for upgrade headroom:
		#   radar          : module_radar_2  (default _3 — room to _3/_4)
		#   vls SAM aux_3  : module_vls_sam_1 (default _2 — room to _2/_3)
		#   ASM aux_2      : module_asm_2  (default _3 — room to _3)

			name = "Choe Hyon Class"

			}

		# Spawn 5 of each hull directly into the NKO fleet. Using

			}
